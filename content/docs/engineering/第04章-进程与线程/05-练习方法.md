## 练习方法

本章的练习方法按照**基础理解→动手实操→问题排查→性能优化→综合设计**五个层级递进设计，每个练习都围绕进程与线程的具体知识点展开，附带完整的代码示例和验证方法。建议读者按照顺序逐步完成，每个练习完成后对照检查标准确认掌握程度。

### 练习一：基础概念理解（预计30分钟）

**目标**：
理解进程与线程的核心概念，能够通过Linux命令行工具观察进程状态、线程结构和系统调用行为，建立从理论到系统的直觉。

**步骤**：

1. **观察进程状态机**（10分钟）

   使用 `ps` 和 `/proc` 文件系统观察真实的进程状态：

   ```bash
   # 查看所有进程的状态字段（STAT列）
   ps -eo pid,ppid,stat,comm | head -30
   # STAT列的含义：S=可中断睡眠, R=运行/就绪, D=不可中断睡眠, Z=僵尸, T=停止

   # 查看一个特定进程的详细状态信息
   cat /proc/1/status | grep -E 'Name|State|Pid|PPid|Threads'
   # State行会显示：sleeping (S)、running (R)、zombie (Z) 等

   # 创建一个僵尸进程并观察
   python3 -c "
   import os, time
   pid = os.fork()
   if pid == 0:
       os._exit(0)  # 子进程立即退出
   else:
       # 父进程不调用 wait()，子进程变成僵尸
       print(f'子进程 PID={pid} 已退出，等待观察僵尸状态...')
       time.sleep(30)  # 30秒内用 ps 观察
   " &amp;
   # 在另一个终端运行：
   # ps -eo pid,ppid,stat,comm | grep defunct
   # 会看到一个 Z (zombie) 状态的进程
   ```

   **验证要点**：
   - 在 `STAT` 列中识别出 `Z`（僵尸）、`S`（睡眠）、`D`（不可中断睡眠）等状态
   - 理解 `D` 状态通常与磁盘 I/O 等待有关，`S` 状态通常与等待事件有关
   - 僵尸进程的特点：占用 PID 表项但不消耗 CPU 和内存

2. **观察线程结构**（10分钟）

   使用工具查看进程内部的线程组织：

   ```bash
   # 启动一个有多线程的程序（如 nginx 或 Chrome 的某个进程）
   # 查看系统中线程数最多的进程
   ps -eo pid,nlwp,comm --sort=-nlwp | head -10
   # nlwp = Number of Lightweight Processes（轻量级进程/线程数）

   # 查看某个进程的所有线程
   ps -T -p <pid>   # -T 显示线程
   # 或
   ls /proc/<pid>/task/   # 每个子目录代表一个线程

   # 查看线程的内核栈和用户栈信息
   cat /proc/<pid>/task/<tid>/stat | awk '{print "线程ID:"$2, "状态:"$3, "内核栈:"$28}'
   ```

   **动手实践**：编写一个C程序，创建4个线程，然后用上述工具观察：

   ```c
   // thread_obs.c - 创建多线程程序并观察
   #include <pthread.h>
   #include <stdio.h>
   #include <unistd.h>

   void *thread_func(void *arg) {
       int id = *(int *)arg;
       printf("线程 %d: PID=%d, TID=%d\n", id, getpid(), gettid());
       sleep(60);  // 保持运行以便观察
       return NULL;
   }

   int main() {
       pthread_t threads[4];
       int ids[4] = {1, 2, 3, 4};
       printf("主线程: PID=%d, TID=%d\n", getpid(), gettid());
       for (int i = 0; i < 4; i++) {
           pthread_create(&amp;threads[i], NULL, thread_func, &amp;ids[i]);
       }
       for (int i = 0; i < 4; i++) {
           pthread_join(threads[i], NULL);
       }
       return 0;
   }
   // 编译运行：
   // gcc -pthread thread_obs.c -o thread_obs &amp;&amp; ./thread_obs &amp;
   // ps -T -p <pid>   观察5个线程（主线程 + 4个工作线程）
   ```

3. **用strace观察fork/exec/wait系统调用**（10分钟）

   ```bash
   # 跟踪一个简单shell命令的系统调用序列
   strace -f -e trace=clone,fork,exec,wait bash -c "ls /tmp"
   # -f: 跟踪子进程
   # 你会看到：fork() → execve("/bin/ls") → wait4() 的完整流程

   # 观察fork的COW效果：跟踪页错误
   strace -e trace=mmap,mprotect,clone python3 -c "
   import os
   pid = os.fork()
   if pid == 0:
       data = bytearray(1024*1024)  # 1MB数据
       data[0] = 42  # 写操作触发COW
       os._exit(0)
   os.wait()
   "
   ```

**检查标准**：
- [ ] 能够从 `ps` 的 STAT 列中识别并解释每种进程状态
- [ ] 能够使用 `/proc` 和 `ps -T` 观察进程中的线程结构
- [ ] 能够用 `strace` 追踪 fork/exec/wait 的系统调用序列
- [ ] 能够解释为什么僵尸进程不消耗 CPU 但需要被回收

### 练习二：动手实操——竞态条件与互斥锁（预计60分钟）

**目标**：
独立编写多线程程序，亲手观察竞态条件（Race Condition）的产生过程，使用互斥锁（Mutex）和原子操作修复问题，深入理解线程同步的必要性。

**步骤**：

1. **制造一个竞态条件**（15分钟）

   编写一个经典的多线程计数器程序，观察数据竞争：

   ```c
   // race_condition.c - 故意制造竞态条件
   #include <pthread.h>
   #include <stdio.h>
   #include <stdlib.h>

   long counter = 0;  // 全局共享变量
   #define INCREMENTS 1000000

   void *increment(void *arg) {
       for (long i = 0; i < INCREMENTS; i++) {
           counter++;  // 这不是原子操作！包含：读取→加1→写入三步
       }
       return NULL;
   }

   int main() {
       pthread_t t1, t2;
       pthread_create(&amp;t1, NULL, increment, NULL);
       pthread_create(&amp;t2, NULL, increment, NULL);
       pthread_join(t1, NULL);
       pthread_join(t2, NULL);
       printf("期望值: %ld\n", 2 * INCREMENTS);
       printf("实际值: %ld\n", counter);
       printf("丢失:   %ld 次操作\n", 2 * INCREMENTS - counter);
       return 0;
   }
   // 编译：gcc -pthread -O0 race_condition.c -o race_condition
   // 注意：必须用 -O0 关闭优化，否则编译器可能将 counter++ 优化为寄存器操作
   // 多次运行会得到不同的结果，每次丢失的数量都不一样
   ```

   **观察要点**：
   - 期望值为 2000000，但实际值几乎总是小于期望值
   - 每次运行结果不同，这就是竞态条件的典型特征
   - 原因分析：`counter++` 在汇编层分为 `MOV → ADD → MOV` 三步，两个线程的指令可能交错执行

2. **用互斥锁修复**（15分钟）

   ```c
   // mutex_fixed.c - 用互斥锁修复竞态条件
   #include <pthread.h>
   #include <stdio.h>

   long counter = 0;
   pthread_mutex_t lock = PTHREAD_MUTEX_INITIALIZER;
   #define INCREMENTS 1000000

   void *increment(void *arg) {
       for (long i = 0; i < INCREMENTS; i++) {
           pthread_mutex_lock(&amp;lock);
           counter++;       // 临界区：被互斥锁保护
           pthread_mutex_unlock(&amp;lock);
       }
       return NULL;
   }

   int main() {
       pthread_t t1, t2;
       pthread_create(&amp;t1, NULL, increment, NULL);
       pthread_create(&amp;t2, NULL, increment, NULL);
       pthread_join(t1, NULL);
       pthread_join(t2, NULL);
       printf("期望值: %ld\n", 2 * INCREMENTS);
       printf("实际值: %ld\n", counter);
       printf(" mutex 修复后结果正确: %s\n", counter == 2 * INCREMENTS ? "YES" : "NO");
       return 0;
   }
   // 编译：gcc -pthread -O0 mutex_fixed.c -o mutex_fixed
   // 运行结果永远是 2000000
   ```

3. **用原子操作修复并对比性能**（15分钟）

   ```c
   // atomic_fixed.c - 用原子操作修复（无锁方案）
   #include <pthread.h>
   #include <stdio.h>
   #include <stdatomic.h>

   atomic_long counter = ATOMIC_VAR_INIT(0);
   #define INCREMENTS 1000000

   void *increment(void *arg) {
       for (long i = 0; i < INCREMENTS; i++) {
           atomic_fetch_add(&amp;counter, 1);  // 原子加法，硬件保证不可分割
       }
       return NULL;
   }

   int main() {
       pthread_t t1, t2;
       pthread_create(&amp;t1, NULL, increment, NULL);
       pthread_create(&amp;t2, NULL, increment, NULL);
       pthread_join(t1, NULL);
       pthread_join(t2, NULL);
       printf("期望值: %ld\n", 2 * INCREMENTS);
       printf("实际值: %ld\n", (long)atomic_load(&amp;counter));
       return 0;
   }
   // 编译：gcc -pthread -O2 atomic_fixed.c -o atomic_fixed
   // 原子操作比互斥锁快 3-10 倍（无内核态切换开销）
   ```

   **性能对比实验**：用 `time` 命令对比三种方案的执行时间：

   ```bash
   for prog in race_condition mutex_fixed atomic_fixed; do
       echo "=== $prog ==="
       time ./$prog
       echo
   done
   # 观察：
   # - race_condition: 最快但结果错误
   # - mutex_fixed: 较慢（涉及内核futex系统调用）但结果正确
   # - atomic_fixed: 介于两者之间，结果正确且性能良好
   ```

4. **理解条件变量的虚假唤醒**（15分钟）

   ```c
   // condition_variable.c - 条件变量的正确用法
   #include <pthread.h>
   #include <stdio.h>
   #include <unistd.h>

   pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
   pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
   int ready = 0;

   void *producer(void *arg) {
       sleep(1);  // 模拟准备工作
       pthread_mutex_lock(&amp;mutex);
       ready = 1;                       // 修改共享状态
       pthread_cond_signal(&amp;cond);      // 通知消费者
       pthread_mutex_unlock(&amp;mutex);
       printf("[生产者] 数据已就绪\n");
       return NULL;
   }

   void *consumer(void *arg) {
       pthread_mutex_lock(&amp;mutex);
       while (ready == 0) {             // 必须用 while，不能用 if！
           // 原因：虚假唤醒（spurious wakeup）
           // 内核在 pthread_cond_wait 返回时可能没有实际的 signal
           // 如果用 if，虚假唤醒会导致在条件未满足时错误地继续执行
           pthread_cond_wait(&amp;cond, &amp;mutex);
       }
       printf("[消费者] 收到数据，ready=%d\n", ready);
       pthread_mutex_unlock(&amp;mutex);
       return NULL;
   }

   int main() {
       pthread_t prod, cons;
       pthread_create(&amp;cons, NULL, consumer, NULL);
       pthread_create(&amp;prod, NULL, producer, NULL);
       pthread_join(prod, NULL);
       pthread_join(cons, NULL);
       return 0;
   }
   // 编译：gcc -pthread condition_variable.c -o condition_variable
   ```

   **关键知识点**：
   - `pthread_cond_wait` 会原子地释放锁并进入等待状态（不会丢失 signal）
   - 必须用 `while` 循环检查条件，因为存在虚假唤醒（POSIX 标准允许内核实现产生虚假唤醒）
   - 必须在持有锁时调用 `pthread_cond_wait`，否则可能错过通知

**检查标准**：
- [ ] 能够独立复现竞态条件并解释其产生原因
- [ ] 能够正确使用互斥锁保护临界区
- [ ] 能够使用 `atomic` 操作实现无锁同步
- [ ] 理解条件变量 `while` 循环检查的必要性
- [ ] 能够用 `time` 命令量化不同同步方案的性能差异

### 练习三：问题排查——僵尸进程与D状态（预计45分钟）

**目标**：
掌握进程异常状态的排查方法，能够独立定位僵尸进程的成因、D状态进程的根因，并实施有效的修复方案。

**步骤**：

1. **制造并排查僵尸进程**（15分钟）

   ```c
   // zombie_creator.c - 故意创建大量僵尸进程
   #include <unistd.h>
   #include <stdio.h>
   #include <stdlib.h>

   int main() {
       for (int i = 0; i < 5; i++) {
           pid_t pid = fork();
           if (pid == 0) {
               printf("子进程 %d 退出\n", getpid());
               exit(0);  // 子进程退出
           }
           // 父进程不调用 wait()，子进程变成僵尸
       }
       printf("父进程 %d 运行中，不回收子进程...\n", getpid());
       sleep(120);  // 等待观察
       return 0;
   }
   // 编译运行：gcc zombie_creator.c -o zombie_creator &amp;&amp; ./zombie_creator
   ```

   **排查流程**：

   ```bash
   # 第一步：发现僵尸进程
   ps aux | grep 'Z' | grep -v grep
   # 输出示例：user  12345  0.0  0.0      0     0 ?  Z  10:00  0:00 [zombie_creator] <defunct>

   # 第二步：找到僵尸进程的父进程
   ps -o ppid= -p 12345
   # 或者更详细：
   ps -eo pid,ppid,stat,comm | grep 12345

   # 第三步：分析父进程为什么没有回收
   # 检查父进程的代码逻辑是否缺少 wait()/waitpid() 调用
   strace -p <parent_pid> -e trace=wait  # 观察父进程是否调用了 wait

   # 第四步：修复 - 杀掉父进程（子进程会被 init 回收）
   kill <parent_pid>

   # 或者注册 SIGCHLD 信号处理（推荐的编程修复方式）：
   # signal(SIGCHLD, SIG_IGN);  // 忽略信号，内核自动回收
   # 或
   # void handler(int sig) { while(waitpid(-1, NULL, WNOHANG) > 0); }
   # signal(SIGCHLD, handler);
   ```

2. **制造并排查D状态进程**（15分钟）

   ```bash
   # 制造一个处于 D 状态（不可中断睡眠）的进程
   # 方法：写入一个非常慢的存储设备
   python3 -c "
   import os, time
   # 创建一个大量数据写入操作
   with open('/dev/sda', 'wb') as f:  # 需要 root 权限
       for _ in range(100):
           f.write(b'x' * 1024 * 1024)
   " 2>/dev/null

   # 如果没有磁盘权限，可以使用 cgroup 模拟：
   # 创建一个受限的 cgroup，然后在其中运行进程

   # 排查 D 状态：
   # 第一步：找出 D 状态进程
   ps aux | awk '$8 == "D" {print $0}'

   # 第二步：查看 /proc 中的等待信息
   cat /proc/<pid>/wchan    # 显示进程正在等待的内核函数名
   cat /proc/<pid>/stack   # 显示内核调用栈（需要 root）

   # 第三步：分析根因
   # 常见原因：
   # - NFS 挂载超时（网络文件系统无响应）
   # - 磁盘硬件故障
   # - 存储驱动 bug
   # - SAN 存储网络中断

   # 第四步：监控 D 状态进程的持续时间
   while true; do
       ps -eo pid,stat,wchan,etime,comm | awk '$2 ~ /D/ {print}'
       sleep 2
   done
   ```

3. **分析上下文切换问题**（15分钟）

   ```bash
   # 观察系统整体的上下文切换频率
   vmstat 1 5
   # 关注 cs（context switches）列
   # 正常值：< 10000/秒
   # 异常值：> 50000/秒 说明有大量线程竞争

   # 使用 perf 精确分析上下文切换
   sudo perf stat -e context-switches -a sleep 5
   # 会显示 5 秒内的总上下文切换次数

   # 找出触发上下文切换最多的进程
   sudo perf top -e context-switches
   # 或者
   sudo perf record -e context-switches -a -- sleep 10
   sudo perf report
   ```

**检查标准**：
- [ ] 能够独立发现僵尸进程并追溯到父进程
- [ ] 理解僵尸进程的三种修复方式（wait/SIG_IGN/SIGCHLD handler）
- [ ] 能够识别 D 状态进程并通过 `/proc` 分析等待原因
- [ ] 能够使用 `vmstat` 和 `perf` 分析上下文切换频率

### 练习四：性能优化——线程池与锁竞争（预计60分钟）

**目标**：
掌握线程池的设计与实现，理解锁竞争对性能的影响，能够使用无锁技术和CPU亲和性优化并发程序性能。

**步骤**：

1. **实现一个基础线程池**（25分钟）

   ```c
   // threadpool.c - 简单线程池实现
   #include <pthread.h>
   #include <stdio.h>
   #include <stdlib.h>
   #include <unistd.h>

   #define POOL_SIZE 4
   #define TASK_QUEUE_SIZE 100

   // 任务结构
   typedef struct task {
       void (*function)(void *);
       void *arg;
       struct task *next;
   } task_t;

   // 线程池结构
   typedef struct {
       pthread_t threads[POOL_SIZE];
       task_t *task_queue;
       task_t *task_tail;
       int task_count;

       pthread_mutex_t mutex;
       pthread_cond_t not_empty;
       pthread_cond_t not_full;
       int shutdown;
   } threadpool_t;

   // 工作线程函数
   void *worker(void *arg) {
       threadpool_t *pool = (threadpool_t *)arg;
       while (1) {
           pthread_mutex_lock(&amp;pool->mutex);
           // 等待任务到来或关闭信号
           while (pool->task_count == 0 &amp;&amp; !pool->shutdown) {
               pthread_cond_wait(&amp;pool->not_empty, &amp;pool->mutex);
           }
           if (pool->shutdown &amp;&amp; pool->task_count == 0) {
               pthread_mutex_unlock(&amp;pool->mutex);
               break;
           }
           // 取出任务
           task_t *task = pool->task_queue;
           pool->task_queue = task->next;
           if (pool->task_queue == NULL) pool->task_tail = NULL;
           pool->task_count--;
           pthread_cond_signal(&amp;pool->not_full);
           pthread_mutex_unlock(&amp;pool->mutex);

           // 执行任务（在锁外面执行，避免阻塞其他线程）
           task->function(task->arg);
           free(task);
       }
       return NULL;
   }

   // 提交任务
   void threadpool_submit(threadpool_t *pool, void (*func)(void *), void *arg) {
       pthread_mutex_lock(&amp;pool->mutex);
       while (pool->task_count >= TASK_QUEUE_SIZE) {
           pthread_cond_wait(&amp;pool->not_full, &amp;pool->mutex);
       }
       task_t *task = malloc(sizeof(task_t));
       task->function = func;
       task->arg = arg;
       task->next = NULL;
       if (pool->task_tail) pool->task_tail->next = task;
       else pool->task_queue = task;
       pool->task_tail = task;
       pool->task_count++;
       pthread_cond_signal(&amp;pool->not_empty);
       pthread_mutex_unlock(&amp;pool->mutex);
   }

   // 初始化线程池
   threadpool_t *threadpool_create() {
       threadpool_t *pool = calloc(1, sizeof(threadpool_t));
       pthread_mutex_init(&amp;pool->mutex, NULL);
       pthread_cond_init(&amp;pool->not_empty, NULL);
       pthread_cond_init(&amp;pool->not_full, NULL);
       for (int i = 0; i < POOL_SIZE; i++) {
           pthread_create(&amp;pool->threads[i], NULL, worker, pool);
       }
       return pool;
   }

   // 关闭线程池
   void threadpool_destroy(threadpool_t *pool) {
       pthread_mutex_lock(&amp;pool->mutex);
       pool->shutdown = 1;
       pthread_cond_broadcast(&amp;pool->not_empty);
       pthread_mutex_unlock(&amp;pool->mutex);
       for (int i = 0; i < POOL_SIZE; i++) {
           pthread_join(pool->threads[i], NULL);
       }
       pthread_mutex_destroy(&amp;pool->mutex);
       pthread_cond_destroy(&amp;pool->not_empty);
       pthread_cond_destroy(&amp;pool->not_full);
       free(pool);
   }

   // 测试任务
   void test_task(void *arg) {
       int id = *(int *)arg;
       printf("任务 %d 在线程 %lu 中执行\n", id, (unsigned long)pthread_self());
       usleep(100000);  // 模拟 100ms 工作
   }

   int main() {
       threadpool_t *pool = threadpool_create();
       // 提交 20 个任务，线程池只有 4 个线程
       for (int i = 0; i < 20; i++) {
           int *id = malloc(sizeof(int));
           *id = i;
           threadpool_submit(pool, test_task, id);
       }
   ```

2. **测量锁竞争的性能影响**（20分钟）

   ```c
   // lock_contention.c - 量化锁竞争的性能开销
   #include <pthread.h>
   #include <stdio.h>
   #include <stdlib.h>
   #include <time.h>

   #define ITERATIONS 10000000
   long shared_counter = 0;
   pthread_mutex_t lock = PTHREAD_MUTEX_INITIALIZER;

   void *increment_with_lock(void *arg) {
       for (long i = 0; i < ITERATIONS; i++) {
           pthread_mutex_lock(&amp;lock);
           shared_counter++;
           pthread_mutex_unlock(&amp;lock);
       }
       return NULL;
   }

   int main() {
       struct timespec start, end;
       int thread_counts[] = {1, 2, 4, 8};
       int num_tests = sizeof(thread_counts) / sizeof(thread_counts[0]);

       for (int t = 0; t < num_tests; t++) {
           int nthreads = thread_counts[t];
           shared_counter = 0;
           pthread_t *threads = malloc(nthreads * sizeof(pthread_t));

           clock_gettime(CLOCK_MONOTONIC, &amp;start);
           for (int i = 0; i < nthreads; i++) {
               pthread_create(&amp;threads[i], NULL, increment_with_lock, NULL);
           }
           for (int i = 0; i < nthreads; i++) {
               pthread_join(threads[i], NULL);
           }
           clock_gettime(CLOCK_MONOTONIC, &amp;end);

           double elapsed = (end.tv_sec - start.tv_sec) +
                           (end.tv_nsec - start.tv_nsec) / 1e9;
           long total_ops = (long)nthreads * ITERATIONS;
           printf("线程数: %2d | 总耗时: %.3fs | 吞吐量: %.0f ops/s\n",
                  nthreads, elapsed, total_ops / elapsed);
           free(threads);
       }
       return 0;
   }
   // 编译：gcc -pthread -O2 lock_contention.c -o lock_contention
   // 运行结果分析：
   // - 1个线程：基准性能
   // - 2个线程：性能可能不升反降（锁竞争抵消了并行收益）
   // - 4个线程：更严重的锁竞争
   // - 8个线程：可能进一步恶化
   // 这就是 Amdahl 定律的体现：串行部分（临界区）限制了并行加速比
   ```

   ```c
   #include <pthread.h>
   #include <sched.h>
   #include <stdio.h>
   #include <unistd.h>

   void set_cpu_affinity(int cpu_id) {
       cpu_set_t cpuset;
       CPU_ZERO(&amp;cpuset);
       CPU_SET(cpu_id, &amp;cpuset);
       pthread_setaffinity_np(pthread_self(), sizeof(cpu_set_t), &amp;cpuset);

       set_cpu_affinity(cpu_id);

       // 验证绑定是否生效
       cpu_set_t current;
       pthread_getaffinity_np(pthread_self(), sizeof(cpu_set_t), &amp;current);
       for (int i = 0; i < 64; i++) {
           if (CPU_ISSET(i, &amp;current)) {
               printf("线程运行在 CPU %d 上\n", i);
           }
       }

       // 模拟CPU密集型计算
       volatile long sum = 0;
       for (long i = 0; i < 1000000000L; i++) {
           sum += i;
       }
       return NULL;
   }

   int main() {
       pthread_t t1, t2;
       int cpu0 = 0, cpu1 = 1;
       pthread_create(&amp;t1, NULL, cpu_intensive_task, &amp;cpu0);
       pthread_create(&amp;t2, NULL, cpu_intensive_task, &amp;cpu1);
       pthread_join(t1, NULL);
   ```

### 练习五：综合设计——生产者-消费者IPC对比（预计90分钟）

**目标**：
设计并实现一个完整的生产者-消费者系统，分别使用管道、共享内存、消息队列三种IPC方式实现，对比不同IPC机制在吞吐量、延迟和编程复杂度上的差异。

**步骤**：

1. **需求分析与架构设计**（15分钟）

   明确系统需求：

   | 指标 | 要求 |
   |------|------|
   | 生产者数量 | 3个进程 |
   | 消费者数量 | 2个进程 |
   | 消息格式 | 结构体：{id, timestamp, payload[256]} |
   | 消息总量 | 100000条 |
   | 性能指标 | 吞吐量(msg/s)、平均延迟(μs) |

   三种IPC方案的技术选型：

   | IPC方式 | 带宽 | 延迟 | 同步需求 | 编程复杂度 |
   |---------|------|------|----------|-----------|
   | 管道(Pipe) | 中 | 低 | 内核自动 | 低 |
   | 共享内存(Shm) | 极高 | 极低 | 需自行实现 | 高 |
   | 消息队列(MQ) | 中 | 中 | 内核提供 | 中 |

2. **管道实现**（20分钟）

   ```c
   // pipe_producer_consumer.c
   #include <stdio.h>
   #include <stdlib.h>
   #include <unistd.h>
   #include <sys/time.h>
   #include <string.h>

   typedef struct {
       int id;
       double timestamp;
       char payload[256];
   } message_t;

   #define NUM_MESSAGES 100000
   #define NUM_PRODUCERS 3
   #define NUM_CONSUMERS 2

   void producer(int pipe_fd, int producer_id) {
       for (int i = 0; i < NUM_MESSAGES / NUM_PRODUCERS; i++) {
           message_t msg;
           msg.id = producer_id * 100000 + i;
           struct timeval tv;
           gettimeofday(&amp;tv, NULL);
           msg.timestamp = tv.tv_sec + tv.tv_usec / 1e6;
           snprintf(msg.payload, 256, "Producer-%d Message-%d", producer_id, i);
           write(pipe_fd, &amp;msg, sizeof(msg));  // 管道是字节流
       }
       close(pipe_fd);
   }

   void consumer(int pipe_fd, int consumer_id, int *count) {
       message_t msg;
       while (read(pipe_fd, &amp;msg, sizeof(msg)) > 0) {
           (*count)++;
       }
   }

   int main() {
       int pipefd[2];
       pipe(pipefd);

       struct timeval start, end;
       gettimeofday(&amp;start, NULL);

       // 创建生产者和消费者进程
       pid_t pids[NUM_PRODUCERS + NUM_CONSUMERS];
       for (int i = 0; i < NUM_PRODUCERS; i++) {
           if (fork() == 0) {
               close(pipefd[0]);  // 子进程关闭读端
               producer(pipefd[1], i);
               exit(0);
           }
       }
       // 所有生产者写完后关闭写端
       // (简化处理，实际需要更复杂的同步)
       close(pipefd[1]);

       int total = 0;
       for (int i = 0; i < NUM_CONSUMERS; i++) {
           pid_t pid = fork();
           if (pid == 0) {
               close(pipefd[1]);
               int count = 0;
               consumer(pipefd[0], i, &amp;count);
               printf("消费者 %d 处理了 %d 条消息\n", i, count);
               exit(0);
           }
       }
       close(pipefd[0]);

       for (int i = 0; i < NUM_PRODUCERS + NUM_CONSUMERS; i++) {
           wait(&amp;pids[i]);
       }

       gettimeofday(&amp;end, NULL);
       double elapsed = (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1e6;
       printf("[管道] 总耗时: %.3fs, 吞吐量: %.0f msg/s\n",
              elapsed, NUM_MESSAGES / elapsed);
       return 0;
   }
   // 编译：gcc pipe_producer_consumer.c -o pipe_pc
   ```

3. **共享内存实现**（25分钟）

   ```c
   // shm_producer_consumer.c
   #include <stdio.h>
   #include <stdlib.h>
   #include <unistd.h>
   #include <sys/shm.h>
   #include <sys/sem.h>
   #include <sys/time.h>
   #include <string.h>

   typedef struct {
       int id;
       double timestamp;
       char payload[256];
   } message_t;

   // 环形缓冲区
   typedef struct {
       message_t buffer[1024];
       int head;     // 写位置
       int tail;     // 读位置
       int count;    // 当前消息数
   } ring_buffer_t;

   #define BUFFER_SIZE 1024
   #define NUM_MESSAGES 100000

   // 信号量操作宏
   void sem_wait_op(int semid, int sem_num) {
       struct sembuf op = {sem_num, -1, 0};
       semop(semid, &amp;op, 1);
   }
   void sem_signal_op(int semid, int sem_num) {
       struct sembuf op = {sem_num, 1, 0};
       semop(semid, &amp;op, 1);
   }

   int main() {
       // 创建共享内存
       int shm_id = shmget(IPC_PRIVATE, sizeof(ring_buffer_t), 0666 | IPC_CREAT);
       ring_buffer_t *rb = (ring_buffer_t *)shmat(shm_id, NULL, 0);
       rb->head = 0; rb->tail = 0; rb->count = 0;

       // 创建信号量：0=mutex, 1=empty_slots, 2=full_slots
       int sem_id = semget(IPC_PRIVATE, 3, 0666 | IPC_CREAT);
       semctl(sem_id, 0, SETVAL, 1);            // mutex = 1
       semctl(sem_id, 1, SETVAL, BUFFER_SIZE);  // empty = BUFFER_SIZE
       semctl(sem_id, 2, SETVAL, 0);            // full = 0

       struct timeval start, end;
       gettimeofday(&amp;start, NULL);

       pid_t pid;
       // 生产者
       for (int i = 0; i < 3; i++) {
           if ((pid = fork()) == 0) {
               for (int j = 0; j < NUM_MESSAGES / 3; j++) {
                   sem_wait_op(sem_id, 1);  // 等待空位
                   sem_wait_op(sem_id, 0);  // 获取互斥锁

                   message_t msg;
                   msg.id = i * 100000 + j;
                   struct timeval tv;
                   gettimeofday(&amp;tv, NULL);
                   msg.timestamp = tv.tv_sec + tv.tv_usec / 1e6;
                   snprintf(msg.payload, 256, "Shm msg %d", j);
                   rb->buffer[rb->head] = msg;
                   rb->head = (rb->head + 1) % BUFFER_SIZE;
                   rb->count++;

                   sem_signal_op(sem_id, 0);  // 释放互斥锁
                   sem_signal_op(sem_id, 2);  // 通知有数据
               }
               exit(0);
           }
       }

       // 消费者
       for (int i = 0; i < 2; i++) {
           if ((pid = fork()) == 0) {
               int count = 0;
               while (count < NUM_MESSAGES / 2) {
                   sem_wait_op(sem_id, 2);  // 等待数据
                   sem_wait_op(sem_id, 0);  // 获取互斥锁

                   message_t msg = rb->buffer[rb->tail];
                   rb->tail = (rb->tail + 1) % BUFFER_SIZE;
                   rb->count--;

                   sem_signal_op(sem_id, 0);  // 释放互斥锁
                   sem_signal_op(sem_id, 1);  // 通知有空位
                   count++;
               }
               printf("[共享内存] 消费者 %d 处理了 %d 条消息\n", i, count);
               exit(0);
           }
       }

       // 等待所有子进程
       for (int i = 0; i < 5; i++) wait(NULL);

       gettimeofday(&amp;end, NULL);
       double elapsed = (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1e6;
       printf("[共享内存] 总耗时: %.3fs, 吞吐量: %.0f msg/s\n",
              elapsed, NUM_MESSAGES / elapsed);

       // 清理
       shmdt(rb);
       shmctl(shm_id, IPC_RMID, NULL);
       semctl(sem_id, 0, IPC_RMID);
       return 0;
   }
   // 编译：gcc shm_producer_consumer.c -o shm_pc
   ```
