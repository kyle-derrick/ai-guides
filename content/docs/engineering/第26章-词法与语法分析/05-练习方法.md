# 词法与语法分析：练习方法

理论再扎实，不动手就无法真正掌握编译器前端技术。本节设计了一套由浅入深、层层递进的练习体系，覆盖词法分析、语法分析、工具链使用、错误处理和性能优化五大维度。每个练习都有明确的目标、可执行的步骤、真实可运行的代码，以及自检标准。

## 练习一：手工实现一个完整的词法分析器（预计 3 小时）

### 目标

从零用 Python 手工实现一个支持多 Token 类型的词法分析器，深入理解有限状态机在词法分析中的实际运作方式。完成后你将能够：设计正则模式、实现 DFA 驱动的扫描器、处理边界情况（转义字符、错误恢复）。

### 步骤一：定义 Token 类型和数据结构（20 分钟）

首先明确你的词法分析器需要识别哪些 Token。以一个简化的 C 语言子集为例：

```python
from enum import Enum, auto
from dataclasses import dataclass
from typing import Optional


class TokenType(Enum):
    """Token 类型枚举"""
    # 关键字
    IF = auto()
    ELSE = auto()
    WHILE = auto()
    RETURN = auto()
    INT = auto()        # int 类型关键字
    FLOAT_KW = auto()   # float 类型关键字

    # 标识符和字面量
    IDENTIFIER = auto()
    INT_LITERAL = auto()
    FLOAT_LITERAL = auto()
    STRING_LITERAL = auto()

    # 运算符
    PLUS = auto()       # +
    MINUS = auto()      # -
    STAR = auto()       # *
    SLASH = auto()      # /
    ASSIGN = auto()     # =
    EQ = auto()         # ==
    NEQ = auto()        # !=
    LT = auto()         # <
    GT = auto()         # >
    LE = auto()         # <=
    GE = auto()         # >=
    NOT = auto()        # !
    AND = auto()        # &amp;&amp;
    OR = auto()         # ||

    # 分隔符
    LPAREN = auto()     # (
    RPAREN = auto()     # )
    LBRACE = auto()     # {
    RBRACE = auto()     # }
    SEMICOLON = auto()  # ;
    COMMA = auto()      # ,

    # 特殊
    NEWLINE = auto()
    EOF = auto()
    ERROR = auto()


@dataclass
class Token:
    """词法单元"""
    type: TokenType
    value: str
    line: int
    column: int

    def __repr__(self):
        return f"Token({self.type.name}, '{self.value}', L{self.line}:C{self.column})"
```

**设计要点**：每个 Token 必须携带行号和列号信息。这不是可选项——当你在后续练习中实现错误报告时，没有位置信息就无法告诉用户"第 3 行第 12 列有语法错误"。

### 步骤二：实现核心扫描器（60 分钟）

词法分析器的核心是一个读取字符、识别模式、输出 Token 的循环。关键技巧是**前瞻（lookahead）**——有些 Token 需要多看一个字符才能确定类型：

```python
class Lexer:
    """手工实现的词法分析器"""

    # 关键字映射表：标识符匹配后查询此表判断是否为关键字
    KEYWORDS = {
        'if': TokenType.IF, 'else': TokenType.ELSE,
        'while': TokenType.WHILE, 'return': TokenType.RETURN,
        'int': TokenType.INT, 'float': TokenType.FLOAT_KW,
    }

    def __init__(self, source: str, filename: str = "<stdin>"):
        self.source = source
        self.filename = filename
        self.pos = 0
        self.line = 1
        self.column = 1

    def peek(self, offset: int = 0) -> str:
        """查看当前位置（或偏移后）的字符，不消耗"""
        idx = self.pos + offset
        if idx < len(self.source):
            return self.source[idx]
        return '\0'

    def advance(self) -> str:
        """消耗当前位置的字符并前进"""
        ch = self.source[self.pos]
        self.pos += 1
        if ch == '\n':
            self.line += 1
            self.column = 1
        else:
            self.column += 1
        return ch

    def skip_whitespace(self):
        """跳过空白字符（空格、制表符、回车），但不跳过换行"""
        while self.pos < len(self.source) and self.source[self.pos] in ' \t\r':
            self.advance()

    def skip_line_comment(self):
        """跳过 // 风格的行注释"""
        while self.pos < len(self.source) and self.peek() != '\n':
            self.advance()

    def skip_block_comment(self):
        """跳过 /* ... */ 风格的块注释"""
        start_line = self.line
        self.advance()  # 跳过 *
        self.advance()  # 跳过 /
        while self.pos < len(self.source):
            if self.peek() == '*' and self.peek(1) == '/':
                self.advance()  # 跳过 *
                self.advance()  # 跳过 /
                return
            self.advance()
        raise LexicalError(
            f"{self.filename}:{start_line}: 未闭合的块注释"
        )

    def read_string(self) -> Token:
        """读取字符串字面量，处理转义字符"""
        start_line, start_col = self.line, self.column
        quote = self.advance()  # 消耗开始的引号
        result = []
        while self.pos < len(self.source):
            ch = self.peek()
            if ch == quote:
                self.advance()  # 消耗结束的引号
                return Token(TokenType.STRING_LITERAL, ''.join(result),
                            start_line, start_col)
            elif ch == '\\':
                self.advance()  # 消耗反斜杠
                esc = self.advance()
                escape_map = {'n': '\n', 't': '\t', '\\': '\\',
                             '"': '"', "'": "'", '0': '\0'}
                result.append(escape_map.get(esc, esc))
            elif ch == '\n':
                raise LexicalError(
                    f"{self.filename}:{start_line}:{start_col}: "
                    f"字符串未闭合（遇到换行符）"
                )
            else:
                result.append(self.advance())
        raise LexicalError(
            f"{self.filename}:{start_line}:{start_col}: 字符串未闭合（到达文件末尾）"
        )

    def read_number(self) -> Token:
        """读取数字字面量（整数或浮点数）"""
        start_line, start_col = self.line, self.column
        result = []

        # 整数部分
        while self.pos < len(self.source) and self.peek().isdigit():
            result.append(self.advance())

        # 小数部分
        is_float = False
        if self.peek() == '.' and self.peek(1).isdigit():
            is_float = True
            result.append(self.advance())  # 消耗 .
            while self.pos < len(self.source) and self.peek().isdigit():
                result.append(self.advance())

        # 科学计数法
        if self.peek() in ('e', 'E'):
            is_float = True
            result.append(self.advance())
            if self.peek() in ('+', '-'):
                result.append(self.advance())
            while self.pos < len(self.source) and self.peek().isdigit():
                result.append(self.advance())

        tok_type = TokenType.FLOAT_LITERAL if is_float else TokenType.INT_LITERAL
        return Token(tok_type, ''.join(result), start_line, start_col)

    def read_identifier(self) -> Token:
        """读取标识符或关键字"""
        start_line, start_col = self.line, self.column
        result = []
        while (self.pos < len(self.source) and
               (self.peek().isalnum() or self.peek() == '_')):
            result.append(self.advance())
        word = ''.join(result)
        tok_type = self.KEYWORDS.get(word, TokenType.IDENTIFIER)
        return Token(tok_type, word, start_line, start_col)

    def next_token(self) -> Token:
        """获取下一个 Token —— 词法分析的主入口"""
        self.skip_whitespace()

        if self.pos >= len(self.source):
            return Token(TokenType.EOF, '', self.line, self.column)

        ch = self.peek()
        start_line, start_col = self.line, self.column

        # 换行
        if ch == '\n':
            self.advance()
            return Token(TokenType.NEWLINE, '\\n', start_line, start_col)

        # 字符串
        if ch in ('"', "'"):
            return self.read_string()

        # 数字
        if ch.isdigit():
            return self.read_number()

        # 标识符或关键字
        if ch.isalpha() or ch == '_':
            return self.read_identifier()

        # 单行注释
        if ch == '/' and self.peek(1) == '/':
            self.skip_line_comment()
            return self.next_token()  # 递归获取下一个实际 Token

        # 块注释
        if ch == '/' and self.peek(1) == '*':
            self.skip_block_comment()
            return self.next_token()

        # 双字符运算符（需要前瞻）
        if ch == '=' and self.peek(1) == '=':
            self.advance(); self.advance()
            return Token(TokenType.EQ, '==', start_line, start_col)
        if ch == '!' and self.peek(1) == '=':
            self.advance(); self.advance()
            return Token(TokenType.NEQ, '!=', start_line, start_col)
        if ch == '<' and self.peek(1) == '=':
            self.advance(); self.advance()
            return Token(TokenType.LE, '<=', start_line, start_col)
        if ch == '>' and self.peek(1) == '=':
            self.advance(); self.advance()
            return Token(TokenType.GE, '>=', start_line, start_col)
        if ch == '&amp;' and self.peek(1) == '&amp;':
            self.advance(); self.advance()
            return Token(TokenType.AND, '&amp;&amp;', start_line, start_col)
        if ch == '|' and self.peek(1) == '|':
            self.advance(); self.advance()
            return Token(TokenType.OR, '||', start_line, start_col)

        # 单字符运算符和分隔符
        single_char_map = {
            '+': TokenType.PLUS, '-': TokenType.MINUS,
            '*': TokenType.STAR, '/': TokenType.SLASH,
            '=': TokenType.ASSIGN, '<': TokenType.LT, '>': TokenType.GT,
            '!': TokenType.NOT, '(': TokenType.LPAREN, ')': TokenType.RPAREN,
            '{': TokenType.LBRACE, '}': TokenType.RBRACE,
            ';': TokenType.SEMICOLON, ',': TokenType.COMMA,
        }
        if ch in single_char_map:
            self.advance()
            return Token(single_char_map[ch], ch, start_line, start_col)

        # 无法识别的字符
        self.advance()
        raise LexicalError(
            f"{self.filename}:{start_line}:{start_col}: "
            f"无法识别的字符 '{ch}'"
        )

    def tokenize(self) -> list:
        """扫描整个源文件，返回 Token 列表"""
        tokens = []
        while True:
            tok = self.next_token()
            tokens.append(tok)
            if tok.type == TokenType.EOF:
                break
        return tokens
```

### 步骤三：测试与验证（30 分钟）

编写测试用例来验证词法分析器的正确性。**关键测试场景**：

```python
def test_lexer():
    source = '''
    int main() {
        float pi = 3.14;
        if (pi < 3.0) {
            return 0;
        }
        // 行注释
        /* 块注释
           可以跨行 */
        char *s = "hello \\"world\\"";
    }
    '''
    lexer = Lexer(source)
    tokens = lexer.tokenize()

    for tok in tokens:
        print(tok)

    # 验证关键 Token 是否被正确识别
    assert any(t.type == TokenType.INT and t.value == 'int' for t in tokens)
    assert any(t.type == TokenType.FLOAT_LITERAL and t.value == '3.14' for t in tokens)
    assert any(t.type == TokenType.STRING_LITERAL and t.value == 'hello \\"world\\"' for t in tokens)
    assert any(t.type == TokenType.AND for t in tokens)  # &amp;&amp; 运算符

    # 验证错误检测
    error_source = 'int x = "unclosed'
    try:
        Lexer(error_source).tokenize()
        assert False, "应该抛出 LexicalError"
    except LexicalError as e:
        print(f"正确检测到错误: {e}")

    print("所有测试通过!")
```

```lex
%{
/*
 * lexer.l —— C 语言子集的词法分析器
 * 功能：与练习一手工版本等价，但由 Flex 自动生成
 */
#include "parser.tab.h"   /* 由 Bison 生成的 Token 定义 */
#include <string.h>
#include <stdlib.h>

int line_num = 1;
int col_num = 1;

/* 更新行列号的辅助宏 */
#define UPDATE_COL col_num += yyleng
%}

%option noyywrap
%option yylineno

DIGIT       [0-9]
LETTER      [a-zA-Z_]
ID          {LETTER}({LETTER}|{DIGIT})*
INT_LIT     {DIGIT}+
FLOAT_LIT   {DIGIT}+"."{DIGIT}*([eE][+-]?{DIGIT}+)?
WS          [ \t\r]+

%%

 /* ====== 关键字（必须在标识符规则之前） ====== */
"if"            { UPDATE_COL; return IF; }
"else"          { UPDATE_COL; return ELSE; }
"while"         { UPDATE_COL; return WHILE; }
"return"        { UPDATE_COL; return RETURN; }
"int"           { UPDATE_COL; return INT_KW; }
"float"         { UPDATE_COL; return FLOAT_KW; }

 /* ====== 字面量 ====== */
{FLOAT_LIT}     { UPDATE_COL; yylval.fval = atof(yytext); return FLOAT_LIT; }
{INT_LIT}       { UPDATE_COL; yylval.ival = atoi(yytext); return INT_LIT; }

\"([^"\\]|\\.)*\"  {
    /* 去掉两端引号，处理转义字符 */
    UPDATE_COL;
    yylval.sval = strdup(yytext + 1);
    yylval.sval[yyleng - 2] = '\0';
    return STRING_LIT;
}

 /* ====== 标识符（在关键字之后） ====== */
{ID}            { UPDATE_COL; yylval.sval = strdup(yytext); return ID; }

 /* ====== 运算符（按长度递减排列） ====== */
"=="            { UPDATE_COL; return EQ; }
"!="            { UPDATE_COL; return NEQ; }
"<="            { UPDATE_COL; return LE; }
">="            { UPDATE_COL; return GE; }
"&amp;&amp;"            { UPDATE_COL; return AND; }
"||"            { UPDATE_COL; return OR; }
"+"             { UPDATE_COL; return PLUS; }
"-"             { UPDATE_COL; return MINUS; }
"*"             { UPDATE_COL; return STAR; }
"/"             { UPDATE_COL; return SLASH; }
"="             { UPDATE_COL; return ASSIGN; }
"<"             { UPDATE_COL; return LT; }
">"             { UPDATE_COL; return GT; }
"!"             { UPDATE_COL; return NOT; }

 /* ====== 分隔符 ====== */
"("             { UPDATE_COL; return LPAREN; }
")"             { UPDATE_COL; return RPAREN; }
"{"             { UPDATE_COL; return LBRACE; }
"}"             { UPDATE_COL; return RBRACE; }
";"             { UPDATE_COL; return SEMICOLON; }
","             { UPDATE_COL; return COMMA; }

 /* ====== 注释和空白 ====== */
"//".*          { UPDATE_COL; /* 跳过行注释 */ }
"/*"            {
    /* 块注释：逐字符扫描直到遇到 */
    int comment_start = yylineno;
    UPDATE_COL;
    int c;
    while ((c = input()) != 0) {
        if (c == '*') {
            c = input();
            if (c == '/') break;
            unput(c);
        }
        if (c == '\n') line_num++;
    }
}
\n              { line_num++; col_num = 1; }
{WS}            { UPDATE_COL; }

 /* ====== 未知字符 ====== */
.               {
    fprintf(stderr, "Lexical error at line %d, col %d: "
                    "unexpected character '%s'\n",
            line_num, col_num, yytext);
    UPDATE_COL;
}

%%
```

**Flex 规范中的关键设计决策**：

| 决策点 | 选择 | 原因 |
|-------|------|------|
| 关键字 vs 标识符 | 关键字规则在前 | 同长度时先出现的规则优先 |
| `==` vs `=` | 双字符在前 | 最长匹配规则自动处理 |
| 块注释处理 | 手工调用 `input()` | Flex 本身不支持嵌套注释 |
| `%option noyywrap` | 启用 | 不需要多文件输入的 `yywrap` 机制 |

## 练习三：实现递归下降语法分析器（预计 4 小时）

### 目标

基于练习一/二产出的 Token 流，用递归下降法实现一个完整的语法分析器，构建表达式和语句的抽象语法树（AST）。这是理解语法分析算法"如何将线性 Token 流组织为层次结构"的最佳实践。

### 步骤一：设计 AST 节点类型（20 分钟）

```python
from dataclasses import dataclass, field
from typing import List, Optional, Union


@dataclass
class ASTNode:
    """AST 节点基类"""
    line: int = 0
    column: int = 0


# ====== 表达式节点 ======

@dataclass
class NumberLiteral(ASTNode):
    value: float = 0
    is_float: bool = False

@dataclass
class StringLiteral(ASTNode):
    value: str = ""

@dataclass
class Identifier(ASTNode):
    name: str = ""

@dataclass
class BinaryOp(ASTNode):
    op: str = ""
    left: Optional[ASTNode] = None
    right: Optional[ASTNode] = None

@dataclass
class UnaryOp(ASTNode):
    op: str = ""
    operand: Optional[ASTNode] = None

@dataclass
class Assignment(ASTNode):
    name: Optional[ASTNode] = None
    value: Optional[ASTNode] = None

@dataclass
class FunctionCall(ASTNode):
    name: Optional[ASTNode] = None
    args: List[ASTNode] = field(default_factory=list)


# ====== 语句节点 ======

@dataclass
class VarDeclaration(ASTNode):
    var_type: str = ""        # "int" / "float"
    name: str = ""
    init: Optional[ASTNode] = None

@dataclass
class IfStatement(ASTNode):
    condition: Optional[ASTNode] = None
    then_body: List[ASTNode] = field(default_factory=list)
    else_body: Optional[List[ASTNode]] = None

@dataclass
class WhileStatement(ASTNode):
    condition: Optional[ASTNode] = None
    body: List[ASTNode] = field(default_factory=list)

@dataclass
class ReturnStatement(ASTNode):
    value: Optional[ASTNode] = None

@dataclass
class Block(ASTNode):
    statements: List[ASTNode] = field(default_factory=list)

@dataclass
class FunctionDef(ASTNode):
    return_type: str = ""
    name: str = ""
    params: List[tuple] = field(default_factory=list)  # [(type, name), ...]
    body: Optional[Block] = None

@dataclass
class Program(ASTNode):
    functions: List[FunctionDef] = field(default_factory=list)
```

**设计要点**：AST 节点要同时承载结构信息和位置信息。位置信息在报错时不可或缺。

### 步骤二：实现语法分析器核心（90 分钟）

递归下降分析器的核心模式是：**每个文法产生式对应一个解析函数**。关键挑战是处理运算符优先级和左递归。

```python
class Parser:
    """
    递归下降语法分析器
    文法（消除左递归后）：
        program     → function_def*
        function_def→ type ID '(' params ')' block
        block       → '{' statement* '}'
        statement   → var_decl | if_stmt | while_stmt | return_stmt
                    | assignment | expr_stmt
        var_decl    → type ID ('=' expr)? ';'
        if_stmt     → 'if' '(' expr ')' block ('else' block)?
        while_stmt  → 'while' '(' expr ')' block
        return_stmt → 'return' expr? ';'
        assignment  → ID '=' expr ';'
        expr_stmt   → expr ';'

        expr        → term (('+' | '-') term)*
        term        → factor (('*' | '/') factor)*
        factor      → NUMBER | STRING | ID ('(' args ')')?
                    | '(' expr ')' | ('+' | '-') factor
    """

    def __init__(self, tokens: List[Token]):
        self.tokens = [t for t in tokens if t.type != TokenType.NEWLINE]
        self.pos = 0

    def current(self) -> Token:
        if self.pos < len(self.tokens):
            return self.tokens[self.pos]
        return Token(TokenType.EOF, '', 0, 0)

    def eat(self, expected: TokenType) -> Token:
        """消费当前 Token，类型不匹配则报错"""
        tok = self.current()
        if tok.type != expected:
            raise SyntaxError(
                f"Line {tok.line}, Col {tok.column}: "
                f"Expected {expected.name}, got {tok.type.name}('{tok.value}')"
            )
        self.pos += 1
        return tok

    def peek_type(self) -> TokenType:
        return self.current().type

    # ====== 语句级解析 ======

    def parse_program(self) -> Program:
        """program → function_def*"""
        program = Program()
        while self.peek_type() != TokenType.EOF:
            program.functions.append(self.parse_function_def())
        return program

    def parse_function_def(self) -> FunctionDef:
        """function_def → type ID '(' params ')' block"""
        tok = self.current()
        ret_type = self.eat_type_keyword()
        name = self.eat(TokenType.IDENTIFIER).value
        self.eat(TokenType.LPAREN)
        params = self.parse_params()
        self.eat(TokenType.RPAREN)
        body = self.parse_block()
        func = FunctionDef(
            return_type=ret_type, name=name,
            params=params, body=body
        )
        func.line = tok.line
        func.column = tok.column
        return func

    def parse_params(self) -> list:
        """params → (type ID (',' type ID)*)?"""
        params = []
        if self.peek_type() in (TokenType.INT, TokenType.FLOAT_KW):
            ptype = self.eat_type_keyword()
            pname = self.eat(TokenType.IDENTIFIER).value
            params.append((ptype, pname))
            while self.peek_type() == TokenType.COMMA:
                self.eat(TokenType.COMMA)
                ptype = self.eat_type_keyword()
                pname = self.eat(TokenType.IDENTIFIER).value
                params.append((ptype, pname))
        return params

    def parse_block(self) -> Block:
        """block → '{' statement* '}'"""
        self.eat(TokenType.LBRACE)
        stmts = []
        while self.peek_type() != TokenType.RBRACE:
            stmts.append(self.parse_statement())
        self.eat(TokenType.RBRACE)
        return Block(statements=stmts)

    def parse_statement(self) -> ASTNode:
        """根据当前 Token 预测语句类型"""
        tok_type = self.peek_type()

        if tok_type in (TokenType.INT, TokenType.FLOAT_KW):
            return self.parse_var_decl()
        elif tok_type == TokenType.IF:
            return self.parse_if_stmt()
        elif tok_type == TokenType.WHILE:
            return self.parse_while_stmt()
        elif tok_type == TokenType.RETURN:
            return self.parse_return_stmt()
        elif tok_type == TokenType.IDENTIFIER:
            # 预测：IDENTIFIER 后面是 '=' 还是 '(' 还是运算符
            # 如果是 ID = expr; → 赋值语句
            # 否则 → 表达式语句（如函数调用）
            if self.peek(1).type == TokenType.ASSIGN:
                return self.parse_assignment()
            else:
                return self.parse_expr_statement()
        else:
            return self.parse_expr_statement()

    def parse_var_decl(self) -> VarDeclaration:
        """var_decl → type ID ('=' expr)? ';'"""
        tok = self.current()
        vtype = self.eat_type_keyword()
        name = self.eat(TokenType.IDENTIFIER).value
        init = None
        if self.peek_type() == TokenType.ASSIGN:
            self.eat(TokenType.ASSIGN)
            init = self.parse_expression()
        self.eat(TokenType.SEMICOLON)
        return VarDeclaration(
            var_type=vtype, name=name, init=init,
            line=tok.line, column=tok.column
        )

    def parse_if_stmt(self) -> IfStatement:
        """if_stmt → 'if' '(' expr ')' block ('else' block)?"""
        tok = self.eat(TokenType.IF)
        self.eat(TokenType.LPAREN)
        cond = self.parse_expression()
        self.eat(TokenType.RPAREN)
        then_body = self.parse_block()
        else_body = None
        if self.peek_type() == TokenType.ELSE:
            self.eat(TokenType.ELSE)
            else_body = self.parse_block()
        return IfStatement(
            condition=cond, then_body=then_body.statements,
            else_body=else_body.statements if else_body else None,
            line=tok.line, column=tok.column
        )

    def parse_while_stmt(self) -> WhileStatement:
        """while_stmt → 'while' '(' expr ')' block"""
        tok = self.eat(TokenType.WHILE)
        self.eat(TokenType.LPAREN)
        cond = self.parse_expression()
        self.eat(TokenType.RPAREN)
        body = self.parse_block()
        return WhileStatement(
            condition=cond, body=body.statements,
            line=tok.line, column=tok.column
        )

    def parse_return_stmt(self) -> ReturnStatement:
        """return_stmt → 'return' expr? ';'"""
        tok = self.eat(TokenType.RETURN)
        value = None
        if self.peek_type() != TokenType.SEMICOLON:
            value = self.parse_expression()
        self.eat(TokenType.SEMICOLON)
        return ReturnStatement(value=value, line=tok.line, column=tok.column)

    def parse_assignment(self) -> Assignment:
        """assignment → ID '=' expr ';'"""
        name_tok = self.eat(TokenType.IDENTIFIER)
        self.eat(TokenType.ASSIGN)
        value = self.parse_expression()
        self.eat(TokenType.SEMICOLON)
        return Assignment(
            name=Identifier(name=name_tok.value, line=name_tok.line),
            value=value, line=name_tok.line, column=name_tok.column
        )

    def parse_expr_statement(self) -> ASTNode:
        """expr_stmt → expr ';'"""
        expr = self.parse_expression()
        self.eat(TokenType.SEMICOLON)
        return expr

    # ====== 表达式级解析（处理优先级） ======

    def parse_expression(self) -> ASTNode:
        """expr → term (('+' | '-') term)*"""
        left = self.parse_term()
        while self.peek_type() in (TokenType.PLUS, TokenType.MINUS):
            op_tok = self.current()
            self.pos += 1
            right = self.parse_term()
            left = BinaryOp(
                op=op_tok.value, left=left, right=right,
                line=op_tok.line, column=op_tok.column
            )
        return left

    def parse_term(self) -> ASTNode:
        """term → factor (('*' | '/') factor)*"""
        left = self.parse_factor()
        while self.peek_type() in (TokenType.STAR, TokenType.SLASH):
            op_tok = self.current()
            self.pos += 1
            right = self.parse_factor()
            left = BinaryOp(
                op=op_tok.value, left=left, right=right,
                line=op_tok.line, column=op_tok.column
            )
        return left

    def parse_factor(self) -> ASTNode:
        """factor → atom | ('+' | '-') factor"""
        # 一元运算符
        if self.peek_type() in (TokenType.PLUS, TokenType.MINUS):
            op_tok = self.current()
            self.pos += 1
            operand = self.parse_factor()
            return UnaryOp(
                op=op_tok.value, operand=operand,
                line=op_tok.line, column=op_tok.column
            )
        return self.parse_atom()

    def parse_atom(self) -> ASTNode:
        """atom → NUMBER | STRING | ID ('(' args ')')? | '(' expr ')'"""
        tok = self.current()

        if tok.type == TokenType.INT_LITERAL:
            self.pos += 1
            return NumberLiteral(value=float(tok.value), is_float=False,
                               line=tok.line, column=tok.column)

        elif tok.type == TokenType.FLOAT_LITERAL:
            self.pos += 1
            return NumberLiteral(value=float(tok.value), is_float=True,
                               line=tok.line, column=tok.column)

        elif tok.type == TokenType.STRING_LITERAL:
            self.pos += 1
            return StringLiteral(value=tok.value, line=tok.line, column=tok.column)

        elif tok.type == TokenType.IDENTIFIER:
            self.pos += 1
            # 检查是否是函数调用
            if self.peek_type() == TokenType.LPAREN:
                self.eat(TokenType.LPAREN)
                args = []
                if self.peek_type() != TokenType.RPAREN:
                    args.append(self.parse_expression())
                    while self.peek_type() == TokenType.COMMA:
                        self.eat(TokenType.COMMA)
                        args.append(self.parse_expression())
                self.eat(TokenType.RPAREN)
                return FunctionCall(
                    name=Identifier(name=tok.value, line=tok.line),
                    args=args, line=tok.line, column=tok.column
                )
            return Identifier(name=tok.value, line=tok.line, column=tok.column)

        elif tok.type == TokenType.LPAREN:
            self.eat(TokenType.LPAREN)
            expr = self.parse_expression()
            self.eat(TokenType.RPAREN)
            return expr

        else:
            raise SyntaxError(
                f"Line {tok.line}, Col {tok.column}: "
                f"Unexpected token {tok.type.name}('{tok.value}')"
            )

    # ====== 辅助方法 ======

    def peek(self, offset: int = 1) -> Token:
        """前瞻：查看当前位置之后第 offset 个 Token"""
        idx = self.pos + offset
        if idx < len(self.tokens):
            return self.tokens[idx]
        return Token(TokenType.EOF, '', 0, 0)

    def eat_type_keyword(self) -> str:
        """消费类型关键字，返回类型名"""
        tok = self.current()
        if tok.type == TokenType.INT:
            self.pos += 1; return "int"
        elif tok.type == TokenType.FLOAT_KW:
            self.pos += 1; return "float"
        raise SyntaxError(
            f"Line {tok.line}: Expected type keyword, got {tok.type.name}"
        )
```

### 步骤三：实现 AST 打印器并验证（45 分钟）

```python
def print_ast(node: ASTNode, indent: int = 0):
    """以缩进树形结构打印 AST，用于验证解析结果"""
    prefix = "  " * indent
    if isinstance(node, Program):
        print(f"{prefix}Program")
        for func in node.functions:
            print_ast(func, indent + 1)
    elif isinstance(node, FunctionDef):
        print(f"{prefix}FunctionDef({node.return_type} {node.name})")
        for ptype, pname in node.params:
            print(f"{prefix}  Param: {ptype} {pname}")
        print_ast(node.body, indent + 1)
    elif isinstance(node, Block):
        print(f"{prefix}Block")
        for stmt in node.statements:
            print_ast(stmt, indent + 1)
    elif isinstance(node, VarDeclaration):
        print(f"{prefix}VarDecl({node.var_type} {node.name})")
        if node.init:
            print(f"{prefix}  Init:", end=" ")
            print_ast(node.init, 0)
    elif isinstance(node, BinaryOp):
        print(f"{prefix}BinaryOp('{node.op}')")
        print_ast(node.left, indent + 1)
        print_ast(node.right, indent + 1)
    elif isinstance(node, NumberLiteral):
        print(f"{prefix}Number({node.value})")
    elif isinstance(node, Identifier):
        print(f"{prefix}Ident({node.name})")
    elif isinstance(node, IfStatement):
        print(f"{prefix}If")
        print(f"{prefix}  Condition:", end=" ")
        print_ast(node.condition, 0)
        print(f"{prefix}  Then:")
        for s in node.then_body:
            print_ast(s, indent + 2)
        if node.else_body:
            print(f"{prefix}  Else:")
            for s in node.else_body:
                print_ast(s, indent + 2)
    else:
        print(f"{prefix}{type(node).__name__}")
```

**第一步：符号表实现**（45 分钟）

```python
class SymbolTable:
    """嵌套作用域的符号表"""

    def __init__(self):
        self.scopes = [{}]  # 全局作用域

    def push_scope(self):
        self.scopes.append({})

    def pop_scope(self):
        if len(self.scopes) > 1:
            self.scopes.pop()

    def define(self, name: str, value, var_type: str = "auto"):
        self.scopes[-1][name] = {'value': value, 'type': var_type}

    def lookup(self, name: str):
        for scope in reversed(self.scopes):
            if name in scope:
                return scope[name]
        raise RuntimeError(f"Undefined variable: {name}")

    def assign(self, name: str, value):
        for scope in reversed(self.scopes):
            if name in scope:
                scope[name]['value'] = value
                return
        raise RuntimeError(f"Undefined variable: {name}")
```

**第二步：解释器实现**（90 分钟）

```python
class Interpreter:
    """AST 解释器"""

    def __init__(self):
        self.global_scope = SymbolTable()
        self.functions = {}  # name → FunctionDef AST node

    def run(self, program: Program):
        # 注册所有函数
        for func in program.functions:
            self.functions[func.name] = func

        # 执行 main 函数（如果存在）
        if 'main' in self.functions:
            self.exec_function('main', [])

    def exec_function(self, name: str, args: list):
        func = self.functions[name]
        old_scope = self.global_scope
        self.global_scope = SymbolTable()

        # 绑定参数
        for i, (ptype, pname) in enumerate(func.params):
            self.global_scope.define(pname, args[i], ptype)

        result = None
        try:
            self.exec_block(func.body)
        except ReturnException as e:
            result = e.value

        self.global_scope = old_scope
        return result

    def exec_block(self, block: Block):
        self.global_scope.push_scope()
        try:
            for stmt in block.statements:
                self.exec_statement(stmt)
        finally:
            self.global_scope.pop_scope()

    def exec_statement(self, stmt: ASTNode):
        if isinstance(stmt, VarDeclaration):
            value = self.eval(stmt.init) if stmt.init else 0
            self.global_scope.define(stmt.name, value, stmt.var_type)
        elif isinstance(stmt, IfStatement):
            cond = self.eval(stmt.condition)
            if cond:
                for s in stmt.then_body:
                    self.exec_statement(s)
            elif stmt.else_body:
                for s in stmt.else_body:
                    self.exec_statement(s)
        elif isinstance(stmt, WhileStatement):
            while self.eval(stmt.condition):
                for s in stmt.body:
                    self.exec_statement(s)
        elif isinstance(stmt, ReturnStatement):
            value = self.eval(stmt.value) if stmt.value else None
            raise ReturnException(value)
        elif isinstance(stmt, Assignment):
            name = stmt.name.name
            value = self.eval(stmt.value)
            self.global_scope.assign(name, value)
        else:
            self.eval(stmt)  # 表达式语句（如函数调用）

    def eval(self, node: ASTNode):
        if isinstance(node, NumberLiteral):
            return node.value
        elif isinstance(node, StringLiteral):
            return node.value
        elif isinstance(node, Identifier):
            return self.global_scope.lookup(node.name)['value']
        elif isinstance(node, BinaryOp):
            left = self.eval(node.left)
            right = self.eval(node.right)
            ops = {
                '+': lambda a, b: a + b,
                '-': lambda a, b: a - b,
                '*': lambda a, b: a * b,
                '/': lambda a, b: a / b if b != 0 else ZeroDivisionError("Division by zero"),
                '==': lambda a, b: a == b,
                '!=': lambda a, b: a != b,
                '<': lambda a, b: a < b,
                '>': lambda a, b: a > b,
                '<=': lambda a, b: a <= b,
                '>=': lambda a, b: a >= b,
                '&amp;&amp;': lambda a, b: a and b,
                '||': lambda a, b: a or b,
            }
            return ops[node.op](left, right)
        elif isinstance(node, UnaryOp):
            operand = self.eval(node.operand)
            return -operand if node.op == '-' else operand
        elif isinstance(node, FunctionCall):
            name = node.name.name
            args = [self.eval(arg) for arg in node.args]
            return self.exec_function(name, args)


class ReturnException(Exception):
    """用于实现 return 语句的控制流异常"""
    def __init__(self, value=None):
        self.value = value
```

## 附录：常见问题与调试技巧

### 问题一：词法分析器吞掉了不该跳过的字符

**症状**：某些 Token 没有被识别，调试发现字符被跳过。

**排查步骤**：
1. 在每个 `advance()` 调用处添加日志，记录消耗的字符
2. 检查 `skip_whitespace()` 是否跳过了非空白字符
3. 检查注释处理是否越界（`/*` 后面的 `*/` 没找到时的边界行为）

### 问题二：语法分析器无限递归

**症状**：程序卡死，栈溢出。

**根因**：文法中存在左递归未消除，如 `expr → expr + term`。

**修复**：改写为循环形式：`expr → term ('+' term)*`，或者提取递归到循环的模式。检查每个递归调用是否最终能到达终止条件（`factor` 不再调用自身）。

### 问题三：Bison shift/reduce 冲突导致语法错误

**症状**：Bison 输出 .warning: 2 shift/reduce conflicts，某些输入解析失败。

**排查步骤**：
1. 查看 `parser.output` 文件中标记了 conflict 的状态
2. 检查是否缺少 `%left`/`%right` 声明
3. 检查文法是否在某个产生式上有歧义
4. 使用 `bison -v` 生成详细的状态机报告

### 问题四：Token 位置信息不准确

**症状**：错误信息指向错误的行号或列号。

**排查步骤**：
1. 确认 `column` 计数在 `advance()` 中正确更新
2. 确认多字符 Token（如 `==`）的位置记录的是开始位置而非结束位置
3. 确认跳过空白和注释后行列号更新正确
