### 1.1 观察进程地址空间布局（15 分钟）

**任务：** 编写一个 C 程序，分别在不同内存区域分配变量，然后通过 `/proc/self/maps` 观察它们在虚拟地址空间中的位置。

```c
// addr_layout.c — 观察不同内存区域的虚拟地址
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

// 全局变量（已初始化）→ 数据段
int global_init = 42;
// 全局变量（未初始化）→ BSS 段
int global_uninit;

int main() {
    // 栈上的局部变量
    int stack_var = 1;
    
    // 堆上分配
    int *heap_var = malloc(sizeof(int));
    *heap_var = 2;
    
    // mmap 匿名映射
    void *mmap_var = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                          MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    
    printf("=== 内存地址观察 ===\n");
    printf("代码段 (main 函数):   %p\n", (void *)main);
    printf("全局已初始化:         %p (值=%d)\n", &amp;global_init, global_init);
    printf("全局未初始化 (BSS):   %p (值=%d)\n", &amp;global_uninit, global_uninit);
    printf("堆变量 (malloc):      %p (值=%d)\n", heap_var, *heap_var);
    printf("栈变量:               %p (值=%d)\n", &amp;stack_var, stack_var);
    printf("mmap 匿名映射:        %p\n", mmap_var);
    
    printf("\n=== 地址空间布局 (/proc/self/maps) ===\n");
    // 打印 maps 文件的前 20 行
    FILE *f = fopen("/proc/self/maps", "r");
    char line[256];
    int count = 0;
    while (fgets(line, sizeof(line), f) &amp;&amp; count < 20) {
        printf("%s", line);
        count++;
    }
    fclose(f);
    
    // 清理
    free(heap_var);
    munmap(mmap_var, 4096);
    return 0;
}
```

### 1.2 页表遍历与 TLB 观察（15 分钟）

**任务：** 通过内核接口观察页表结构和 TLB 行为。

```bash
# 方法一：使用 pagemap 查看虚拟页到物理页的映射
# 需要 root 权限
sudo cat /proc/$(pidof addr_layout)/pagemap | head -20
# 每个条目 8 字节，包含：
# bit 0-54: 物理页帧号（PFN）
# bit 63: 页面是否在物理内存中（present 位）

# 方法二：使用 Linux 内置的 page-types 工具
# 查看特定地址的页面类型
python3 -c "
import struct
pid = $(pidof addr_layout)
with open(f'/proc/{pid}/pagemap', 'rb') as f:
    # 读取前 10 个页面的 pagemap 条目
    for i in range(10):
        entry = struct.unpack('Q', f.read(8))[0]
        present = (entry >> 63) &amp; 1
        swapped = (entry >> 62) &amp; 1
       PFN = entry &amp; 0x7FFFFFFFFFFFFF
        print(f'VPN {i}: present={present}, swapped={swapped}, PFN={PFN}')
"
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/mman.h>

#define PAGE_SIZE 4096
#define NUM_PAGES (1024 * 1024)  // 100 万个页面 = 4GB

int main() {
    // 方案一：用 4KB 页面映射 4GB 内存
    size_t size_4k = (size_t)NUM_PAGES * PAGE_SIZE;
    char *ptr_4k = mmap(NULL, size_4k, PROT_READ | PROT_WRITE,
                        MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    
    // 依次访问每个页面的首个字节（触发 TLB 填充）
    for (int i = 0; i < NUM_PAGES; i += 1024) {
        ptr_4k[i * PAGE_SIZE] = 1;
    }
    munmap(ptr_4k, size_4k);
    
    return 0;
}
```

```bash
# HugePages_Total:    1024
# HugePages_Free:     1024
```

```c
// hugepage_bench.c — 大页 vs 标准页性能对比
#include <stdio.h>
#include <stdlib.h>
#include <sys/mman.h>
#include <time.h>

#define REGION_SIZE (256 * 1024 * 1024)  // 256MB
#define STRIDE 4096                       // 每次访问跳一个页面

static double elapsed_ms(struct timespec start, struct timespec end) {
    return (end.tv_sec - start.tv_sec) * 1000.0 +
           (end.tv_nsec - start.tv_nsec) / 1e6;
}

void benchmark(const char *label, void *addr, size_t size) {
    struct timespec t0, t1;
    volatile char sum = 0;
    int iterations = 10;
    
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    for (int iter = 0; iter < iterations; iter++) {
        for (size_t off = 0; off < size; off += STRIDE) {
            sum += ((volatile char *)addr)[off];
        }
    }
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    
    printf("%s: %.2f ms (sum=%d)\n", label, elapsed_ms(t0, t1), sum);
}

int main() {
    // 标准 4KB 页面
    void *std = mmap(NULL, REGION_SIZE, PROT_READ | PROT_WRITE,
                     MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    if (std == MAP_FAILED) { perror("mmap std"); return 1; }
    
    // 2MB 大页
    void *hp = mmap(NULL, REGION_SIZE, PROT_READ | PROT_WRITE,
                    MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB, -1, 0);
    if (hp == MAP_FAILED) {
        perror("mmap hugepage (需要 root 或足够大页)");
        munmap(std, REGION_SIZE);
        return 1;
    }
    
    // 预触摸所有页面（触发缺页）
    for (size_t i = 0; i < REGION_SIZE; i += 4096)
        ((volatile char *)std)[i] = 1;
    for (size_t i = 0; i < REGION_SIZE; i += 4096)
        ((volatile char *)hp)[i] = 1;
    
    // 交替运行以消除缓存影响
    benchmark("标准 4KB 页面", std, REGION_SIZE);
    benchmark("2MB 大页    ", hp, REGION_SIZE);
    benchmark("标准 4KB 页面", std, REGION_SIZE);
    benchmark("2MB 大页    ", hp, REGION_SIZE);
    
    munmap(std, REGION_SIZE);
    munmap(hp, REGION_SIZE);
    return 0;
}
```

```bash
sudo ./hugepage_bench

# 差异来源：大页减少了 TLB miss，从而减少了页表遍历开销

```

### 2.1 跟踪 malloc 的底层行为（20 分钟）

**任务：** 使用 strace 和 /proc 观察 malloc 在不同分配大小下的底层行为（brk vs mmap）。

```c
// malloc_trace.c — 观察不同大小分配的底层路径
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

void print_heap_info(const char *label) {
    char cmd[256];
    snprintf(cmd, sizeof(cmd), "cat /proc/%d/maps | grep -c '\\[heap\\]'", getpid());
    FILE *f = popen(cmd, "r");
    int heap_count;
    fscanf(f, "%d", &amp;heap_count);
    pclose(f);
    
    // 获取堆的大小
    FILE *maps = fopen("/proc/self/maps", "r");
    char line[256];
    while (fgets(line, sizeof(line), maps)) {
        if (strstr(line, "[heap]")) {
            unsigned long start, end;
            sscanf(line, "%lx-%lx", &amp;start, &amp;end);
            printf("%s: 堆范围 %lx-%lx (大小 %.1f KB)\n",
                   label, start, end, (end - start) / 1024.0);
        }
    }
    fclose(maps);
}

int main() {
    printf("=== malloc 底层行为观察 ===\n\n");
    
    // 观察 1：小分配（走 brk/sbrk 扩展堆）
    printf("--- 小分配 (1KB) ---\n");
    print_heap_info("分配前");
    void *p1 = malloc(1024);
    print_heap_info("分配后");
    *(int *)p1 = 42;
    printf("分配 1KB: 指针=%p\n\n", p1);
    
    // 观察 2：中等分配（通常走 mmap）
    printf("--- 中等分配 (128KB) ---\n");
    void *p2 = malloc(128 * 1024);
    printf("分配 128KB: 指针=%p (可能走 mmap，检查 maps 中是否有 [anon] 区域)\n", p2);
    
    // 观察 3：大分配（一定走 mmap）
    printf("--- 大分配 (1MB) ---\n");
    void *p3 = malloc(1024 * 1024);
    printf("分配 1MB: 指针=%p\n", p3);
    
    // 观察 4：mmap 阈值（ptmalloc2 默认 128KB）
    printf("\n--- ptmalloc2 mmap 阈值 ---\n");
    printf("glibc malloc 在分配 >= 128KB 时默认使用 mmap 而非 brk\n");
    printf("可通过 M_MMAP_THRESHOLD mallopt 调整\n");
    
    free(p1);
    free(p2);
    free(p3);
    return 0;
}
```

### 2.2 写时复制（COW）行为观察（20 分钟）

**任务：** 编写程序观察 fork() 后 COW 的实际行为——父子进程共享物理页面，直到一方写入。

```c
// cow_demo.c — 写时复制行为观察
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/mman.h>
#include <sys/wait.h>

#define SIZE (1024 * 1024)  // 1MB

void show_rss(const char *label) {
    char cmd[256];
    snprintf(cmd, sizeof(cmd),
        "grep VmRSS /proc/%d/status | awk '{print \"%s: RSS = \" $2 \" KB\"}'",
        getpid(), label);
    system(cmd);
}

int main() {
    // 分配 1MB 内存并写入数据
    char *shared_mem = mmap(NULL, SIZE, PROT_READ | PROT_WRITE,
                            MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    memset(shared_mem, 'A', SIZE);
    
    printf("=== 写时复制（COW）观察 ===\n\n");
    show_rss("父进程（分配后）");
    
    pid_t pid = fork();
    if (pid == 0) {
        // 子进程
        show_rss("子进程（fork 后，未写入）");
        printf("  → 注意：子进程的 RSS 与父进程相近\n");
        printf("  → 但这并不意味着复制了 1MB 物理内存！\n");
        printf("  → 父子进程通过页表共享同一份物理页面（标记为只读）\n");
        
        // 子进程写入第一个页面 → 触发 COW
        shared_mem[0] = 'B';
        show_rss("子进程（写入 1 字节后）");
        printf("  → RSS 几乎不变：只复制了被写入的那个页面（4KB）\n");
        
        // 子进程写入整个区域 → 大量 COW
        memset(shared_mem, 'C', SIZE);
        show_rss("子进程（写入全部 1MB 后）");
        printf("  → RSS 增加约 1MB：每个页面都触发了 COW\n");
        
        munmap(shared_mem, SIZE);
        exit(0);
    } else {
        wait(NULL);
        show_rss("父进程（子进程退出后）");
        munmap(shared_mem, SIZE);
    }
    
    printf("\n=== 关键结论 ===\n");
    printf("1. fork() 不复制物理内存，只复制页表（标记为 COW）\n");
    printf("2. 只有写入时才复制被写入的页面（粒度 = 1 个页面 = 4KB）\n");
    printf("3. 这使得 fork() 的开销极低，即使进程内存很大\n");
    printf("4. 大量写入会触发大量 COW，显著增加 RSS 和缺页次数\n");
    return 0;
}
```

### 2.3 mmap 文件映射 vs read/write 对比（20 分钟）

**任务：** 编写两个版本的文件读取程序，对比 mmap 映射和传统 read 系统调用的性能差异。

```c
// file_read_bench.c — mmap vs read 性能对比
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <time.h>

#define BLOCK_SIZE (4 * 1024)  // 4KB 读取块

static double elapsed_ms(struct timespec s, struct timespec e) {
    return (e.tv_sec - s.tv_sec) * 1000.0 + (e.tv_nsec - s.tv_nsec) / 1e6;
}

// 方法一：传统 read()
long read_method(const char *path) {
    int fd = open(path, O_RDONLY);
    if (fd < 0) { perror("open"); return -1; }
    
    struct stat sb;
    fstat(fd, &amp;sb);
    long total = 0;
    char buf[BLOCK_SIZE];
    ssize_t n;
    
    while ((n = read(fd, buf, BLOCK_SIZE)) > 0) {
        total += n;
    }
    close(fd);
    return total;
}

// 方法二：mmap 映射
long mmap_method(const char *path) {
    int fd = open(path, O_RDONLY);
    if (fd < 0) { perror("open"); return -1; }
    
    struct stat sb;
    fstat(fd, &amp;sb);
    
    void *mapped = mmap(NULL, sb.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
    if (mapped == MAP_FAILED) { perror("mmap"); close(fd); return -1; }
    
    long total = 0;
    volatile char sum = 0;
    // 逐页访问确保数据实际被读入内存
    for (size_t i = 0; i < sb.st_size; i += BLOCK_SIZE) {
        sum += ((volatile char *)mapped)[i];
        total++;
    }
    
    munmap(mapped, sb.st_size);
    close(fd);
    return total * BLOCK_SIZE;
}

// 方法三：read + madvise（预读优化）
long readahead_method(const char *path) {
    int fd = open(path, O_RDONLY);
    if (fd < 0) { perror("open"); return -1; }
    
    struct stat sb;
    fstat(fd, &amp;sb);
    
    // 提示内核即将顺序读取整个文件
    madvise(NULL, sb.st_size, MADV_SEQUENTIAL);
    
    long total = 0;
    char buf[BLOCK_SIZE];
    ssize_t n;
    while ((n = read(fd, buf, BLOCK_SIZE)) > 0) {
        total += n;
    }
    close(fd);
    return total;
}

int main(int argc, char *argv[]) {
    const char *path = (argc > 1) ? argv[1] : "/usr/lib/x86_64-linux-gnu/libc.so.6";
    struct stat sb;
    stat(path, &amp;sb);
    printf("测试文件: %s (%.1f MB)\n\n", path, sb.st_size / (1024.0 * 1024.0));
    
    struct timespec t0, t1;
    long total;
    int rounds = 5;
    
    // 清除缓存（需要 root）
    printf("提示：运行前执行 'sudo sh -c \"echo 3 > /proc/sys/vm/drop_caches\"' 清除缓存\n\n");
    
    // 测试 read
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    for (int i = 0; i < rounds; i++) total = read_method(path);
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    printf("read():        %.2f ms/轮 (共 %ld 字节)\n",
           elapsed_ms(t0, t1) / rounds, total);
    
    // 测试 mmap
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    for (int i = 0; i < rounds; i++) total = mmap_method(path);
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    printf("mmap():        %.2f ms/轮\n", elapsed_ms(t0, t1) / rounds);
```

```c
// malloc_bench.c — 多线程 malloc 性能基准
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <time.h>

#define NUM_THREADS 4
#define ALLOCS_PER_THREAD 1000000
#define ALLOC_SIZE 64  // 64 字节小对象分配

static double elapsed_ms(struct timespec s, struct timespec e) {
    return (e.tv_sec - s.tv_sec) * 1000.0 + (e.tv_nsec - s.tv_nsec) / 1e6;
}

void *worker(void *arg) {
    int id = *(int *)arg;
    srand(id);
    
    // 混合分配/释放模式
    void *ptrs[1000];
    for (int i = 0; i < ALLOCS_PER_THREAD; i++) {
        int idx = i % 1000;
        if (ptrs[idx]) {
            free(ptrs[idx]);
        }
        ptrs[idx] = malloc(ALLOC_SIZE);
        ((char *)ptrs[idx])[0] = 'x';  // 触发实际页面访问
    }
    // 清理
    for (int i = 0; i < 1000; i++) {
        if (ptrs[i]) free(ptrs[i]);
    }
    return NULL;
}

int main() {
    pthread_t threads[NUM_THREADS];
    int ids[NUM_THREADS];
    
    printf("=== malloc 多线程性能基准 ===\n");
    printf("线程数: %d, 每线程分配次数: %d, 对象大小: %d 字节\n",
           NUM_THREADS, ALLOCS_PER_THREAD, ALLOC_SIZE);
    
    struct timespec t0, t1;
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    
    for (int i = 0; i < NUM_THREADS; i++) {
        ids[i] = i;
        pthread_create(&amp;threads[i], NULL, worker, &amp;ids[i]);
    }
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }
    
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    printf("总耗时: %.2f ms\n", elapsed_ms(t0, t1));
    printf("吞吐量: %.0f allocs/sec\n",
           (double)NUM_THREADS * ALLOCS_PER_THREAD / (elapsed_ms(t0, t1) / 1000.0));
    
    return 0;
}
```

```c
// cache_bench.c — 缓存行效应与内存布局对比
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define N 4096

static double elapsed_ms(struct timespec s, struct timespec e) {
    return (e.tv_sec - s.tv_sec) * 1000.0 + (e.tv_nsec - s.tv_nsec) / 1e6;
}

// 行优先遍历（缓存友好）
void row_major_sum(long matrix[N][N], long *result) {
    long sum = 0;
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            sum += matrix[i][j];
        }
    }
    *result = sum;
}

// 列优先遍历（缓存不友好）
void col_major_sum(long matrix[N][N], long *result) {
    long sum = 0;
    for (int j = 0; j < N; j++) {
        for (int i = 0; i < N; i++) {
            sum += matrix[i][j];
        }
    }
    *result = sum;
}

int main() {
    long (*matrix)[N] = malloc(sizeof(long) * N * N);
    memset(matrix, 1, sizeof(long) * N * N);
    
    struct timespec t0, t1;
    long result;
    
    // 预热：确保数据在缓存中（首次运行时观察冷启动效应）
    row_major_sum(matrix, &amp;result);
    
    // 行优先
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    row_major_sum(matrix, &amp;result);
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    printf("行优先遍历: %.2f ms\n", elapsed_ms(t0, t1));
    
    // 列优先
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    col_major_sum(matrix, &amp;result);
    clock_gettime(CLOCK_MONOTONIC, &amp;t1);
    printf("列优先遍历: %.2f ms\n", elapsed_ms(t0, t1));
    
    printf("\n=== 分析 ===\n");
    printf("行优先 vs 列优先的性能比通常为 3-10 倍\n");
    printf("原因：行优先访问时相邻元素在同一缓存行（64字节），\n");
    printf("      一次内存加载可服务 8 个 long 的读取。\n");
    printf("      列优先每次跳跃 N*8=32KB，几乎每次都 cache miss。\n");
    
    free(matrix);
    return 0;
}
```

```c
// false_sharing.c — 伪共享问题演示
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <time.h>

#define ITERATIONS 100000000

// 问题：两个热更新字段在同一缓存行
struct BadLayout {
    volatile long counter_a;  // 线程 A 更新
    volatile long counter_b;  // 线程 B 更新
    // 两个字段总共 16 字节，在同一 64 字节缓存行中
};

// 解决方案：填充到不同缓存行
struct GoodLayout {
    volatile long counter_a;
    char padding[56];  // 填充 56 字节，确保在不同缓存行
    volatile long counter_b;
};

struct BadLayout bad = {0, 0};
struct GoodLayout good = {0, {0}, 0};

void *increment_bad(void *arg) {
    for (long i = 0; i < ITERATIONS; i++) {
        __sync_fetch_and_add(&amp;bad.counter_a, 1);
    }
    return NULL;
}

void *increment_good(void *arg) {
    for (long i = 0; i < ITERATIONS; i++) {
        __sync_fetch_and_add(&amp;good.counter_a, 1);
    }
    return NULL;
}

int main() {
    pthread_t t1, t2;
    struct timespec t0, t1_time, t2_time;
    
    // 伪共享场景
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    pthread_create(&amp;t1, NULL, increment_bad, NULL);
    pthread_create(&amp;t2, NULL, increment_bad, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    clock_gettime(CLOCK_MONOTONIC, &amp;t1_time);
    printf("伪共享 (BadLayout):   %.2f ms\n",
           (t1_time.tv_sec - t0.tv_sec) * 1000.0 +
           (t1_time.tv_nsec - t0.tv_nsec) / 1e6);
    
    // 无伪共享场景
    clock_gettime(CLOCK_MONOTONIC, &amp;t0);
    pthread_create(&amp;t1, NULL, increment_good, NULL);
    pthread_create(&amp;t2, NULL, increment_good, NULL);
    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    clock_gettime(CLOCK_MONOTONIC, &amp;t2_time);
    printf("无伪共享 (GoodLayout): %.2f ms\n",
           (t2_time.tv_sec - t0.tv_sec) * 1000.0 +
           (t2_time.tv_nsec - t0.tv_nsec) / 1e6);
    
    printf("\n=== 伪共享的代价 ===\n");
    printf("伪共享可导致多线程性能降低 3-10 倍！\n");
    printf("解决方法：\n");
    printf("  1. 填充（padding）：如本例所示\n");
    printf("  2. __attribute__((aligned(64))) 对齐到缓存行\n");
    printf("  3. Java: @sun.misc.Contended 注解\n");
    return 0;
}
```

```bash
# 无伪共享 (GoodLayout): ~300 ms  （快 5 倍）
```

### 5.1 场景：设计一个高并发缓存服务（30 分钟）

**场景描述：** 为一个日活 1000 万的社交平台设计消息缓存服务。要求：

- 服务器配置：双路 Xeon，128GB DDR4，2TB NVMe SSD

**要求产出：**

设计文档应包含以下内容：

1. 硬件层配置

   - THP 模式选择

2. 操作系统层配置
   - vm.swappiness 值及理由
   - vm.overcommit_memory 设置
   - vm.dirty_ratio / vm.dirty_background_ratio
   - cgroup 内存限制

3. 应用层设计
   - 内存分配器选择及理由

4. 监控方案
   - 关键指标及告警阈值
   - 内存泄漏检测策略
   - 性能基线和对比方法

**参考答案要点：**

```bash
# 1. 硬件层
# 大页：100GB 需要 51200 个 2MB 大页
echo 51200 > /proc/sys/vm/nr_hugepages

# NUMA：将缓存进程绑定到一个节点，避免跨节点访问
numactl --cpunodebind=0 --membind=0 cache-server

# Swap：完全禁用（延迟敏感服务）
swapoff -a

# THP：禁用（数据库/缓存服务推荐）
echo never > /sys/kernel/mm/transparent_hugepage/enabled

# 2. 操作系统层
# 减少 Swap 倾向（已禁用 swap，此设置为双保险）
sysctl vm.swappiness=0

# 允许 overcommit（避免 mmap 失败）
sysctl vm.overcommit_memory=1

# 减少脏页刷盘频率（让 OS 有更多 buffer）
sysctl vm.dirty_ratio=40
sysctl vm.dirty_background_ratio=10

# 3. 应用层
# 使用 jemalloc（多线程场景最优）
LD_PRELOAD=/usr/lib/x86_64-linux-gnu/libjemalloc.so ./cache-server

# 或在编译时链接
# gcc -ljemalloc -o cache-server ...

# 数据结构：使用紧凑数组而非链表（减少碎片和指针开销）
# 例如：哈希表用开放寻址而非链表法
```

### 5.2 问题排查实战（30 分钟）

**场景：** 你的缓存服务上线两周后，出现以下问题。请制定排查方案：

问题现象：
1. 内存使用从初始的 90GB 缓慢增长到 115GB（两周增长 25GB）
2. 每天凌晨 2-3 点出现一次 P99 延迟飙升到 5ms
3. `top` 命令显示 RES 列持续增长
4. swapoff 后 free 命令显示可用内存从 30GB 降到 5GB

**排查思路：**

```bash
# 第一步：确认内存增长趋势
# 收集过去两周的内存数据
sar -r -f /var/log/sysstat/sa* | awk '{print $1, $4, $7}' | tail -50
# 确认是否是线性增长（泄漏）还是周期性波动（正常行为）

# 第二步：定位内存消耗来源
# 使用 pmap 分析进程内存组成
pmap -x $(pgrep cache-server) | sort -k3 -n -r | head -20
# 关注最大的匿名映射区域

# 使用 smem 区分共享和独占内存
sudo smem -t -k -p -P cache-server

# 第三步：检查凌晨延迟飙升
# 检查 cron 任务
crontab -l
# 检查是否有定时清理、备份等 IO 密集操作
# 检查内核日志
dmesg -T | grep -i "oom\|kill\|memory\|swap" | tail -20

# 第四步：分析泄漏可能的原因
# 如果是 Java 应用：
jmap -heap $(pgrep cache-server)
jstat -gcutil $(pgrep cache-server) 5000 20
# 观察 Old 区是否持续增长

# 如果是 C/C++ 应用：
# 在测试环境用 Valgrind 运行
valgrind --leak-check=full ./cache-server

# 第五步：凌晨问题分析
# 检查系统级内存回收行为
cat /proc/vmstat | grep -E "^(allocstall|pgscan|pgsteal)"
# 如果 pgscan 持续增长 → 内核在积极回收内存
```
