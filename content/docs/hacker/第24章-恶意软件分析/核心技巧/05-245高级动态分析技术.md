## 24.5 高级动态分析技术

### 24.5.1 调试器高级用法

**x64dbg高级断点技巧**

```text
// 条件断点
// 在特定API上设置条件断点，只在满足条件时中断
// 例如：只在CreateFileW打开特定文件时中断
// 条件表达式：strstr(arg(1), ".txt") != 0

// 日志断点
// 记录API调用参数但不中断执行
// 日志格式：[CreateFileW] {s:[esp+4]}

// 内存断点
// 监控特定内存区域的读/写/执行
// 用于检测数据解密完成的时机

// 硬件断点
// 最多4个硬件断点，不影响程序执行速度
// 用于监控特定地址的访问
```

**Frida高级Hook技术**

```javascript
// Hook Native函数
Interceptor.attach(Module.findExportByName('kernel32.dll', 'CreateFileW'), {
    onEnter: function(args) {
        this.filename = Memory.readUtf16String(args[0]);
        console.log('[CreateFileW] ' + this.filename);
    },
    onLeave: function(retval) {
        if (this.filename.indexOf('.txt') !== -1) {
            console.log('[CreateFileW] Returned: ' + retval);
        }
    }
});

// Hook CryptoAPI
Interceptor.attach(Module.findExportByName('advapi32.dll', 'CryptEncrypt'), {
    onEnter: function(args) {
        console.log('[CryptEncrypt] Key Handle: ' + args[2]);
        console.log('[CryptEncrypt] Final: ' + args[4]);
        
        // 读取加密前的数据
        var dataLen = Memory.readDword(args[5]);
        var data = Memory.readByteArray(args[3], dataLen);
        console.log('[CryptEncrypt] Plaintext: ' + hexdump(data));
    },
    onLeave: function(retval) {
        // 读取加密后的数据
        console.log('[CryptEncrypt] Ciphertext written');
    }
});

// 内存扫描
function scanMemory(pattern) {
    Process.enumerateRanges('r--').forEach(function(range) {
        Memory.scan(range.base, range.size, pattern, {
            onMatch: function(address, size) {
                console.log('[Found] ' + address + ' in ' + range.base);
            }
        });
    });
}

// API跟踪
function traceModule(moduleName) {
    var module = Process.findModuleByName(moduleName);
    module.enumerateExports().forEach(function(exp) {
        if (exp.type === 'function') {
            Interceptor.attach(exp.address, {
                onEnter: function(args) {
                    console.log('[' + exp.name + '] called');
                }
            });
        }
    });
}
```

### 24.5.2 网络协议逆向分析

# 使用Frida解密网络通信
import frida

js_code = """
// Hook SSL_write和SSL_read
Interceptor.attach(Module.findExportByName('ssl.dll', 'SSL_write'), {
    onEnter: function(args) {
        var data = Memory.readByteArray(args[1], parseInt(args[2]));
        console.log('[SSL_write] ' + hexdump(data));
    }
});

Interceptor.attach(Module.findExportByName('ssl.dll', 'SSL_read'), {
    onEnter: function(args) {
        this.buf = args[1];
        this.len = args[2];
    },
    onLeave: function(retval) {
        var data = Memory.readByteArray(this.buf, parseInt(retval));
        console.log('[SSL_read] ' + hexdump(data));
    }
});

// Hook加密函数以获取密钥
Interceptor.attach(Module.findExportByName('crypt32.dll', 'CryptDeriveKey'), {
    onEnter: function(args) {
        console.log('[CryptDeriveKey] Algorithm: ' + Memory.readDword(args[1]));
    }
});
"""

session = frida.attach("target.exe")
script = session.create_script(js_code)
script.load()

**C2协议解析**

```python
# 分析HTTP C2通信模式
import requests
from scapy.all import *

def analyze_http_c2(pcap_file):
    """分析HTTP C2通信模式"""
    packets = rdpcap(pcap_file)
    
    http_requests = []
    for pkt in packets:
        if pkt.haslayer(Raw):
            payload = pkt[Raw].load.decode('utf-8', errors='ignore')
            if 'HTTP' in payload:
                http_requests.append({
                    'src': pkt[IP].src,
                    'dst': pkt[IP].dst,
                    'payload': payload
                })
    
    # 识别C2通信模式
    c2_indicators = []
    for req in http_requests:
        # 检查异常的User-Agent
        if 'User-Agent' in req['payload']:
            ua = re.search(r'User-Agent: (.+)', req['payload'])
            if ua and is_suspicious_ua(ua.group(1)):
                c2_indicators.append('Suspicious User-Agent')
        
        # 检查Cookie中的编码数据
        if 'Cookie' in req['payload']:
            cookie = re.search(r'Cookie: (.+)', req['payload'])
            if cookie and has_encoded_data(cookie.group(1)):
                c2_indicators.append('Encoded Cookie Data')
        
        # 检查POST数据
        if 'POST' in req['payload']:
            body = req['payload'].split('\r\n\r\n')[-1]
            if len(body) > 100 and is_base64(body):
                c2_indicators.append('Base64 POST Data')
    
    return c2_indicators
```

### 24.5.3 内存取证辅助分析

```python
# 在动态分析中使用内存取证技术
import volatility3
from volatility3.framework import contexts
from volatility3.plugins.windows import pslist, netscan, malfind

def analyze_memory_dump(memory_path):
    """分析内存转储"""
    context = contexts.Context()
    
    # 列出所有进程
    for proc in pslist.PsList.list_processes(context, memory_path):
        print(f"PID: {proc.UniqueProcessId}")
        print(f"Name: {proc.ImageFileName}")
        print(f"PPID: {proc.InheritedFromUniqueProcessId}")
        print(f"Create Time: {proc.CreateTime}")
        print()
    
    # 检查网络连接
    for conn in netscan.NetScan.scan(context, memory_path):
        print(f"Local: {conn.LocalAddr}:{conn.LocalPort}")
        print(f"Remote: {conn.ForeignAddr}:{conn.ForeignPort}")
        print(f"State: {conn.State}")
        print()
    
    # 检测进程注入
    for proc, vad_base, data in malfind.Malfind.scan(context, memory_path):
        if is_injected_code(data):
            print(f"[!] Injected code in PID {proc.UniqueProcessId}")
            print(f"    VAD Base: {hex(vad_base)}")
            print(f"    First bytes: {data[:16].hex()}")
```
