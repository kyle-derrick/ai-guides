#### 1.2 实现基础分布式锁类（20分钟）

创建 `distributed_lock.py`，实现以下功能：

```python
import redis
import uuid
import time
import threading
from typing import Optional


class DistributedLock:
    """基于 Redis 的单节点分布式锁"""

    # 释放锁的 Lua 脚本：比较+删除原子执行
    RELEASE_LUA = """
    if redis.call("GET", KEYS[1]) == ARGV[1] then
        return redis.call("DEL", KEYS[1])
    else
        return 0
    end
    """

    # 续期 Lua 脚本：检查+续期原子执行
    RENEW_LUA = """
    if redis.call("GET", KEYS[1]) == ARGV[1] then
        return redis.call("EXPIRE", KEYS[1], ARGV[2])
    else
        return 0
    end
    """

    def __init__(self, redis_client: redis.Redis, lock_name: str,
                 expire_seconds: int = 30):
        self.redis = redis_client
        self.lock_key = f"lock:{lock_name}"
        self.value = str(uuid.uuid4())  # 唯一标识，防止误释放
        self.expire_seconds = expire_seconds
        self._release_script = self.redis.register_script(self.RELEASE_LUA)
        self._renew_script = self.redis.register_script(self.RENEW_LUA)
        self._watchdog: Optional[threading.Timer] = None
        self._locked = False

    def acquire(self, timeout: float = 10.0, retry_interval: float = 0.1) -> bool:
        """
        尝试获取锁，支持自旋重试。
        timeout: 最大等待时间（秒），0 表示立即返回
        retry_interval: 重试间隔（秒）
        """
        deadline = time.monotonic() + timeout
        while True:
            result = self.redis.set(
                self.lock_key, self.value, nx=True, ex=self.expire_seconds
            )
            if result:
                self._locked = True
                self._start_watchdog()
                return True
            if time.monotonic() >= deadline:
                return False
            time.sleep(retry_interval)

    def release(self) -> bool:
        """安全释放锁：只有持有者才能释放"""
        self._stop_watchdog()
        result = self._release_script(
            keys=[self.lock_key], args=[self.value]
        )
        self._locked = False
        return result == 1

    def _start_watchdog(self):
        """启动看门狗线程，每 expire/3 秒续期一次"""
        def _renew():
            if self._locked:
                renewed = self._renew_script(
                    keys=[self.lock_key],
                    args=[self.value, self.expire_seconds]
                )
                if renewed:
                    # 续期成功，安排下一次
                    self._watchdog = threading.Timer(
                        self.expire_seconds / 3, _renew
                    )
                    self._watchdog.daemon = True
                    self._watchdog.start()

        self._watchdog = threading.Timer(
            self.expire_seconds / 3, _renew
        )
        self._watchdog.daemon = True
        self._watchdog.start()

    def _stop_watchdog(self):
        if self._watchdog:
            self._watchdog.cancel()
            self._watchdog = None

    def __enter__(self):
        if not self.acquire():
            raise TimeoutError("获取分布式锁超时")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.release()
        return False
```

**关键设计点（动手前务必理解）：**

| 设计决策 | 原因 |
|---------|------|
| `value` 使用 UUID | 防止客户端A释放客户端B的锁（误区一） |
| 释放使用 Lua 脚本 | GET+DEL 必须原子执行，否则存在 TOCTOU 竞态 |
| 续期使用 Lua 脚本 | 检查值+EXPIRE 必须原子，否则可能续期别人的锁 |
| 看门狗 daemon 线程 | 进程退出时线程随之终止，锁最终过期释放，不会死锁 |
| `__enter__`/`__exit__` | 支持 `with` 语句，保证异常时也能释放锁 |

#### 1.3 编写并发测试（15分钟）

创建 `test_lock.py`，模拟多进程竞争同一把锁：

```python
import redis
import threading
import time
import random
from collections import Counter
from distributed_lock import DistributedLock

r = redis.Redis(host='localhost', port=6379, decode_responses=True)

# 测试1：互斥性验证
def test_mutual_exclusion():
    """验证同一时刻只有一个客户端持有锁"""
    concurrency_violations = []
    shared_counter = {"value": 0}
    lock = DistributedLock(r, "mutex_test", expire_seconds=10)

    def worker(worker_id):
        for _ in range(20):
            if lock.acquire(timeout=5):
                current = shared_counter["value"]
                time.sleep(random.uniform(0.001, 0.01))
                shared_counter["value"] = current + 1
                lock.release()
            else:
                concurrency_violations.append(worker_id)

    threads = [threading.Thread(target=worker, args=(i,)) for i in range(10)]
    for t in threads:
        t.start()
    for t in threads:
        t.join(timeout=30)

    expected = 200  # 10 threads × 20 iterations
    actual = shared_counter["value"]
    print(f"[互斥性测试] 期望计数={expected}, 实际计数={actual}, "
          f"违规次数={len(concurrency_violations)}")
    assert actual == expected, f"互斥性被破坏! expected={expected}, got={actual}"
    print("✅ 互斥性测试通过")


# 测试2：安全释放验证
def test_safe_release():
    """验证客户端A不能释放客户端B的锁"""
    r.delete("lock:safe_release_test")
    lock_a = DistributedLock(r, "safe_release_test", expire_seconds=30)
    lock_b = DistributedLock(r, "safe_release_test", expire_seconds=30)

    # A 获取锁
    assert lock_a.acquire(timeout=1) is True
    # B 尝试获取（应失败）
    assert lock_b.acquire(timeout=0.1) is False
    # B 尝试释放（应无效）
    result = lock_b.release()
    assert result is False, "客户端B不应能释放客户端A的锁"
    # A 正常释放
    result = lock_a.release()
    assert result is True
    print("✅ 安全释放测试通过")


# 测试3：锁续期验证
def test_watchdog_renewal():
    """验证看门狗在业务长时间执行时能续期"""
    r.delete("lock:watchdog_test")
    lock = DistributedLock(r, "watchdog_test", expire_seconds=3)  # 短过期
    assert lock.acquire(timeout=1) is True

    # 业务执行 8 秒，超过 3 秒的过期时间
    # 如果看门狗不工作，锁会过期，其他客户端能获取
    time.sleep(8)

    # 此时如果看门狗正常工作，锁仍应被持有
    lock_b = DistributedLock(r, "watchdog_test", expire_seconds=1)
    got_b = lock_b.acquire(timeout=0.1)
    if not got_b:
        print("✅ 看门狗续期测试通过（8秒后锁仍被持有）")
    else:
        print("❌ 看门狗续期失败（锁已过期）")
    lock.release()


# 测试4：with 语句异常安全性
def test_context_manager():
    """验证异常发生时锁能被正确释放"""
    r.delete("lock:ctx_test")
    try:
        with DistributedLock(r, "ctx_test", expire_seconds=10) as lock:
            raise ValueError("业务异常")
    except ValueError:
        pass

    # 锁应已被释放
    result = r.get("lock:ctx_test")
    print(f"[context_manager测试] 锁值={result}")
    assert result is None, "with 语句异常后锁未释放"
    print("✅ 上下文管理器测试通过")


if __name__ == "__main__":
    test_mutual_exclusion()
    test_safe_release()
    test_watchdog_renewal()
    test_context_manager()
    print("\n🎉 所有基础测试通过!")
```

## 练习二：Lua脚本深度实验——亲手验证原子性（预计40分钟）

### 目标

#### 2.1 设计竞态对照实验（20分钟）

创建 `race_condition_experiment.py`：

```python
import redis
import threading
import time

r = redis.Redis(host='localhost', port=6379, decode_responses=True)

def unsafe_release(client_id):
    """不使用 Lua 脚本的释放：GET + if match + DEL（三步非原子）"""
    key = "lock:race_test"
    my_value = f"client_{client_id}"

    val = r.get(key)
    if val == my_value:
        # ⚠️ 关键：在 GET 和 DEL 之间，其他客户端可能已经获取了锁
        time.sleep(0.005)  # 故意引入延迟放大竞态窗口
        r.delete(key)
        return True
    return False


def safe_release_lua(client_id):
    """使用 Lua 脚本的释放：原子比较+删除"""
    key = "lock:race_test"
    my_value = f"client_{client_id}"
    lua_script = """
    if redis.call("GET", KEYS[1]) == ARGV[1] then
        return redis.call("DEL", KEYS[1])
    else
        return 0
    end
    """
    result = r.eval(lua_script, 1, key, my_value)
    return result == 1


def acquire_lock(client_id, key="lock:race_test"):
    return r.set(key, f"client_{client_id}", nx=True, ex=30)


def release_lock(client_id, use_lua):
    if use_lua:
        return safe_release_lua(client_id)
    else:
        return unsafe_release(client_id)


def run_experiment(use_lua, rounds=100):
    """模拟高并发下的锁释放竞态"""
    violations = 0
    success_count = {"A": 0, "B": 0}

    for _ in range(rounds):
        r.delete("lock:race_test")

        def worker(name, use_lua_flag):
            # 模拟业务执行
            time.sleep(0.002)
            release_lock(name, use_lua_flag)

        # Client A 获取锁
        if not acquire_lock("A"):
            continue
        # Client B 也尝试获取（应失败）
        acquire_lock("B")

        # 并发释放
        t1 = threading.Thread(target=worker, args=("A", use_lua))
        t2 = threading.Thread(target=worker, args=("B", use_lua))
        t1.start()
        t2.start()
        t1.join(timeout=5)
        t2.join(timeout=5)

        # 检查是否出现了误释放：A释放后B还能获取锁吗？
        # 这里检测的是释放过程中的竞态

    return violations


# 实验1：非原子释放（预期出现竞态）
print("=" * 60)
print("实验1：非原子释放 (GET + DEL)")
print("=" * 60)
violations_no_lua = 0
for trial in range(200):
    r.delete("lock:race_test")
    acquire_lock("A", "lock:race_test")

    # 两个线程并发竞争
    result_holder = {"released": False}
    def try_release():
        val = r.get("lock:race_test")
        time.sleep(0.003)  # 竞态窗口
        r.delete("lock:race_test")
        result_holder["released"] = True

    t1 = threading.Thread(target=try_release)
    t2 = threading.Thread(target=try_release)
    t1.start(); t2.start()
    t1.join(5); t2.join(5)

    # 第二个DEL实际已经删了不该删的东西
    # 检测：重新获取锁应该成功（说明确实有竞态窗口）
    # 这个实验的价值在于理解竞态窗口的存在

print(f"非原子释放实验完成（观察竞态窗口）")

# 实验2：Lua脚本原子释放
print("\n" + "=" * 60)
print("实验2：Lua脚本原子释放")
print("=" * 60)
violations_lua = 0
for trial in range(200):
    r.delete("lock:race_test")
    acquire_lock("A", "lock:race_test")
    acquire_lock("B", "lock:race_test")

    def worker_lua(name):
        release_lock(name, use_lua=True)

    t1 = threading.Thread(target=worker_lua, args=("A",))
    t2 = threading.Thread(target=worker_lua, args=("B",))
    t1.start(); t2.start()
    t1.join(5); t2.join(5)

print(f"Lua脚本原子释放实验完成（无竞态）")

print("\n" + "=" * 60)
print("结论：Lua 脚本保证了 GET+DEL 的原子性，")
print("消除了非原子操作中的 TOCTOU (Time-of-Check to Time-of-Use) 竞态。")
```

#### 2.2 编写更复杂的 Lua 脚本（20分钟）

用 Redis 作为练习场，逐步增加 Lua 脚本的复杂度。以下练习要求在 Redis 客户端中直接执行 Lua 脚本：

```bash
# 练习2a：可重入锁的加锁脚本
# 要求：同一客户端可多次加锁（计数+1），不同客户端互斥
cat > reentrant_lock.lua << 'EOF'
-- KEYS[1]: 锁的key
-- ARGV[1]: 客户端标识
-- ARGV[2]: 过期时间（秒）
local key = KEYS[1]
local client = ARGV[1]
local ttl = tonumber(ARGV[2])

-- 锁不存在 → 首次加锁
if redis.call("EXISTS", key) == 0 then
    redis.call("HSET", key, client, 1)
    redis.call("EXPIRE", key, ttl)
    return 1
end

-- 锁已被自己持有 → 重入计数+1
if redis.call("HGET", key, client) then
    redis.call("HINCRBY", key, client, 1)
    redis.call("EXPIRE", key, ttl)
    return 1
end

-- 锁被其他客户端持有 → 加锁失败
return 0
EOF

# 练习2b：可重入锁的释放脚本
cat > reentrant_unlock.lua << 'EOF'
local key = KEYS[1]
local client = ARGV[1]

if not redis.call("HGET", key, client) then
    return 0  -- 不是持有者
end

local count = redis.call("HINCRBY", key, client, -1)
if count <= 0 then
    redis.call("DEL", key)  -- 重入计数归零，删除整个key
end
return 1
EOF

# 练习2c：读写锁的获取脚本（挑战题）
cat > rw_lock_read.lua << 'EOF'
-- KEYS[1]: 读锁key (Hash: client -> count)
-- KEYS[2]: 写锁key (String: client -> 1)
-- ARGV[1]: 客户端标识
-- ARGV[2]: 过期时间
-- 没有写锁时，可以获取读锁
if redis.call("EXISTS", KEYS[2]) == 0 then
    redis.call("HINCRBY", KEYS[1], ARGV[1], 1)
    redis.call("EXPIRE", KEYS[1], ARGV[2])
    return 1
end
return 0
EOF
```

用 Python 逐一验证上述脚本的正确性，构造以下场景并断言结果：

| 场景 | 操作序列 | 预期结果 |
|------|---------|---------|
| 可重入：同一客户端三次加锁 | A→A→A | 三次都成功，Hash value=3 |
| 可重入：释放两次后仍持有 | A→A→A→unlock→unlock | 仍持有（count=1），第三次unlock才真正释放 |
| 可重入：互斥 | A加锁→B加锁 | B失败（返回0） |
| 读写锁：多读并行 | read(A)→read(B) | 两次都成功 |
| 读写锁：读写互斥 | read(A)→write(B) | write失败 |
| 读写锁：写后无读 | write(A)→read(B) | read失败 |

#### 3.2 故障一：锁过期但业务未完成（20分钟）

**故障现象**：两个客户端同时持有锁，数据库出现脏写。

```python
"""
故障模拟：锁过期导致的并发冲突
场景：订单状态机，A正在处理订单A的支付回调，锁过期后B也获取了锁
"""
import redis
import time

r = redis.Redis(host='localhost', port=6380, decode_responses=True)
r.flushall()

LOCK_KEY = "order_lock:ORDER_001"
ORDER_KEY = "order:ORDER_001"


def simulate_fault_scenario():
    """模拟故障：锁3秒过期，但业务执行需要8秒"""

    # 客户端A获取锁
    r.set(LOCK_KEY, "client_A", nx=True, ex=3)  # 3秒过期
    r.set(ORDER_KEY, "CREATED")
    print(f"[T=0s] 客户端A获取锁，订单状态: {r.get(ORDER_KEY)}")

    # 客户端A开始处理... 但业务很慢
    time.sleep(4)  # 4秒后锁已过期（3秒过期）

    # 客户端B也获取到了锁！
    result = r.set(LOCK_KEY, "client_B", nx=True, ex=3)
    if result:
        print(f"[T=4s] ⚠️ 客户端B获取到锁！锁已过期")
        r.set(ORDER_KEY, "PAID_BY_B")  # B设置状态
        print(f"[T=4s] 客户端B将订单状态改为: {r.get(ORDER_KEY)}")

    # 客户端A恢复执行，继续修改状态（以为自己还持有锁）
    r.set(ORDER_KEY, "SHIPPED_BY_A")  # A基于旧状态操作
    print(f"[T=4s] 客户端A将订单状态改为: {r.get(ORDER_KEY)}")

    # 🐛 问题：A在PAID_BY_B基础上直接改成SHIPPED_BY_A
    # 正确流程应该是 PAID → SHIPPED，但B的PAID操作可能不完整
    print(f"\n❌ 最终状态: {r.get(ORDER_KEY)}")
    print("问题：A不知道B已经修改了状态，直接覆盖")


def fix_with_version_check():
    """修复方案：使用乐观锁（版本号）保护状态机转移"""
    r.flushall()
    r.set(ORDER_KEY, "CREATED")
    r.hset("order:ORDER_001:meta", "version", 1)

    # Lua 脚本：状态转移 + 版本号检查（原子操作）
    TRANSITION_LUA = """
    local key = KEYS[1]
    local meta_key = KEYS[2]
```

```python
"""
分布式锁性能基准测试
测试项：加锁延迟、释放延迟、锁竞争吞吐量
"""
import redis
import time
import threading
import statistics
from concurrent.futures import ThreadPoolExecutor, as_completed

r = redis.Redis(host='localhost', port=6379, decode_responses=True, max_connections=20)


def benchmark_single_lock_ops():
    """单线程下的锁操作延迟（无竞争基线）"""
    r.delete("bench:lock")
    results = {"acquire": [], "release": []}

    for _ in range(10000):
        # 加锁
        start = time.perf_counter_ns()
        r.set("bench:lock", "bench_client", nx=True, ex=30)
        end = time.perf_counter_ns()
        results["acquire"].append((end - start) / 1000)  # 转微秒

        # 释放
        start = time.perf_counter_ns()
        r.eval("""
            if redis.call("GET", KEYS[1]) == ARGV[1] then
                return redis.call("DEL", KEYS[1])
            end
            return 0
        """, 1, "bench:lock", "bench_client")
        end = time.perf_counter_ns()
        results["release"].append((end - start) / 1000)

    for op, times in results.items():
        print(f"[{op}] avg={statistics.mean(times):.1f}μs, "
              f"p50={statistics.median(times):.1f}μs, "
              f"p99={sorted(times)[int(len(times)*0.99)]:.1f}μs, "
              f"p999={sorted(times)[int(len(times)*0.999)]:.1f}μs")


def benchmark_contention():
    """多线程竞争同一把锁时的吞吐量"""
    r.delete("bench:contention_lock")
    success_count = [0]
    fail_count = [0]
    lock = threading.Lock()  # Python本地锁保护计数器

    def worker():
        for _ in range(100):
            result = r.set("bench:contention_lock", "x", nx=True, ex=10)
            with lock:
                if result:
                    success_count[0] += 1
                    time.sleep(0.001)  # 模拟业务
                    r.delete("bench:contention_lock")
                else:
                    fail_count[0] += 1

    start = time.perf_counter()
    threads = [threading.Thread(target=worker) for _ in range(50)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    elapsed = time.perf_counter() - start

    print(f"[并发竞争] 50线程×100次 = 5000次请求")
    print(f"  成功: {success_count[0]}, 失败: {fail_count[0]}")
    print(f"  总耗时: {elapsed:.2f}s")
    print(f"  吞吐量: {success_count[0]/elapsed:.1f} ops/s")


def benchmark_different_lock_keys():
    """不同锁粒度对吞吐量的影响"""
    granularities = [
        ("全局锁:global", 1),
        ("类型锁:type_A", 26),
        ("实例锁:inst_", 100),
    ]

    for prefix, count in granularities:
        r.flushdb()
        success = [0]
        lock = threading.Lock()

        def worker(key_name):
            for _ in range(200):
                result = r.set(f"bench:{key_name}", "x", nx=True, ex=5)
                with lock:
                    if result:
                        success[0] += 1
                        r.delete(f"bench:{key_name}")

        start = time.perf_counter()
        threads = []
        for i in range(50):
            key_name = f"{prefix}{i % count}"
            threads.append(threading.Thread(target=worker, args=(key_name,)))
        for t in threads:
            t.start()
        for t in threads:
            t.join()
        elapsed = time.perf_counter() - start

        print(f"[{prefix}] 粒度={count}把锁, "
              f"成功率={success[0]}/{50*200}, "
              f"吞吐量={success[0]/elapsed:.1f} ops/s")


if __name__ == "__main__":
    print("=" * 60)
    print("基准测试1：单线程锁操作延迟")
    print("=" * 60)
    benchmark_single_lock_ops()

    print("\n" + "=" * 60)
    print("基准测试2：多线程锁竞争吞吐量")
    print("=" * 60)
    benchmark_contention()

    print("\n" + "=" * 60)
    print("基准测试3：锁粒度对性能的影响")
    print("=" * 60)
    benchmark_different_lock_keys()
```

#### 4.2 性能调优方向（15分钟）

完成压测后，分析以下优化策略的效果，并用数据验证：

| 优化策略 | 适用场景 | 预期效果 |
|---------|---------|---------|
| **减小锁粒度** | 全局锁→资源实例锁 | 吞吐量提升 10-100 倍 |
| **降低锁持有时间** | 锁内只做必要操作 | 减少竞争等待时间 |
| **使用批量操作** | Redis Pipeline 加锁 | 减少 RTT 次数 |
| **锁升级为乐观锁** | 读多写少场景 | 消除锁等待 |
| **Redis Cluster 分片** | 不同 key 分布在不同 slot | 利用多节点并行 |

```python
"""
优化对比：Pipeline 批量操作 vs 逐条命令
"""
def benchmark_with_pipeline():
    """使用 Pipeline 减少网络往返"""
    r.delete("bench:lock")

    # 逐条命令：1000次操作 = 2000次RTT（加锁+释放各1000次）
    start = time.perf_counter()
    for i in range(1000):
        r.set("bench:lock", f"c{i}", nx=True, ex=30)
        r.eval("""
            if redis.call("GET", KEYS[1]) == ARGV[1] then
                return redis.call("DEL", KEYS[1])
            end
            return 0
        """, 1, "bench:lock", f"c{i}")
    serial_time = time.perf_counter() - start
    print(f"[逐条命令] 1000次操作: {serial_time:.3f}s")

    # Pipeline：1000次操作 = 2次RTT（1000条命令打包发送）
    start = time.perf_counter()
    for i in range(1000):
        pipe = r.pipeline()
        pipe.set("bench:lock", f"c{i}", nx=True, ex=30)
        pipe.eval("""
            if redis.call("GET", KEYS[1]) == ARGV[1] then
                return redis.call("DEL", KEYS[1])
            end
            return 0
        """, 1, "bench:lock", f"c{i]")
        pipe.execute()
    pipeline_time = time.perf_counter() - start
    print(f"[Pipeline] 1000次操作: {pipeline_time:.3f}s")
    print(f"加速比: {serial_time/pipeline_time:.1f}x")
```
