# 练习方法

架构风格的学习不能止步于"知道有哪些风格"——真正的理解来自于动手实践。本章设计了六个递进式练习，从概念辨析到架构决策，从代码实现到架构评审，覆盖"道法术器"四个层面。每个练习都基于真实场景，要求读者不仅仅是画图和写代码，而是像真正的架构师一样思考、权衡和决策。

```python
# domain/user.py - 领域层：纯业务逻辑，不依赖任何外部框架
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class User:
    """用户领域实体 - 包含业务行为，不是贫血模型"""
    id: str
    email: str
    username: str
    created_at: datetime = field(default_factory=datetime.now)
    is_active: bool = True
    failed_login_attempts: int = 0

    # 业务行为 - 这不是贫血模型！
    def deactivate(self):
        """用户主动停用账户"""
        if not self.is_active:
            raise ValueError("用户已经是停用状态")
        self.is_active = False

    def record_failed_login(self, max_attempts: int = 5):
        """记录失败登录，超过阈值自动锁定"""
        self.failed_login_attempts += 1
        if self.failed_login_attempts >= max_attempts:
            self.is_active = False

    def reset_login_attempts(self):
        """登录成功后重置失败计数"""
        self.failed_login_attempts = 0

    def change_email(self, new_email: str):
        """修改邮箱，执行业务规则校验"""
        if not new_email or '@' not in new_email:
            raise ValueError("邮箱格式无效")
        self.email = new_email
```

```python
# domain/user_repository.py - 领域层定义端口（接口）
from abc import ABC, abstractmethod
from typing import Optional
from .user import User

class UserRepository(ABC):
    """仓储接口 - 领域层定义，基础设施层实现
    
    这就是六边形架构的"端口"：
    - 领域层不依赖任何数据库技术
    - 切换数据库只需提供新的适配器实现
    """
    @abstractmethod
    async def find_by_id(self, user_id: str) -> Optional[User]:
        ...

    @abstractmethod
    async def find_by_email(self, email: str) -> Optional[User]:
        ...

    @abstractmethod
    async def save(self, user: User) -> None:
        ...

    @abstractmethod
    async def delete(self, user_id: str) -> None:
        ...
```

```python
# infrastructure/sql_user_repository.py - 基础设施层实现适配器
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from domain.user import User
from domain.user_repository import UserRepository
from infrastructure.models import UserModel  # ORM 模型

class SqlUserRepository(UserRepository):
    """SQLAlchemy 实现的仓储适配器"""
    
    def __init__(self, session: AsyncSession):
        self.session = session

    async def find_by_id(self, user_id: str) -> Optional[User]:
        result = await self.session.execute(
            select(UserModel).where(UserModel.id == user_id)
        )
        model = result.scalar_one_or_none()
        return self._to_domain(model) if model else None

    async def save(self, user: User) -> None:
        model = self._to_model(user)
        self.session.add(model)
        await self.session.commit()

    def _to_domain(self, model: UserModel) -> User:
        """ORM模型 → 领域实体（防腐层）"""
        return User(
            id=model.id,
            email=model.email,
            username=model.username,
            created_at=model.created_at,
            is_active=model.is_active,
            failed_login_attempts=model.failed_login_attempts,
        )

    def _to_model(self, user: User) -> UserModel:
        """领域实体 → ORM模型"""
        return UserModel(
            id=user.id,
            email=user.email,
            username=user.username,
            created_at=user.created_at,
            is_active=user.is_active,
            failed_login_attempts=user.failed_login_attempts,
        )
```

```python
# application/user_service.py - 应用层：编排领域服务，不含业务规则
from domain.user import User
from domain.user_repository import UserRepository
from domain.exceptions import UserNotFoundError, DuplicateEmailError

class ApplicationUserService:
    """应用服务 - 编排用例流程，调用领域层执行业务规则"""
    
    def __init__(self, user_repo: UserRepository):
        self.user_repo = user_repo  # 依赖接口，不依赖实现

    async def register_user(self, email: str, username: str) -> User:
        """注册用户 - 用例编排"""
        # 检查邮箱唯一性（业务规则委托给领域层判断）
        existing = await self.user_repo.find_by_email(email)
        if existing:
            raise DuplicateEmailError(f"邮箱 {email} 已被注册")
        
        # 创建领域实体（业务行为在实体内部）
        user = User(id=generate_id(), email=email, username=username)
        await self.user_repo.save(user)
        return user

    async def handle_login(self, user_id: str) -> None:
        """处理登录 - 编排失败计数逻辑"""
        user = await self.user_repo.find_by_id(user_id)
        if not user:

        user.reset_login_attempts()
        await self.user_repo.save(user)
```

使用 Python 实现一个简化的订单事件驱动系统：

```python
# event_bus.py - 事件总线（最简实现）
import asyncio
from typing import Callable, Dict, List, Any
from dataclasses import dataclass, field
from datetime import datetime
from uuid import uuid4

@dataclass
class Event:
    """事件基类"""
    event_id: str = field(default_factory=lambda: str(uuid4()))
    event_type: str = ""
    timestamp: datetime = field(default_factory=datetime.now)
    payload: Dict[str, Any] = field(default_factory=dict)
    metadata: Dict[str, Any] = field(default_factory=dict)

class EventBus:
    """进程内事件总线 - 支持异步订阅和发布
    
    注意：这是教学用的简化实现。
    生产环境应使用 RabbitMQ、Kafka 等消息中间件。
    """
    def __init__(self):
        self._handlers: Dict[str, List[Callable]] = {}

    def subscribe(self, event_type: str, handler: Callable):
        """订阅事件"""
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)

    async def publish(self, event: Event):
        """发布事件 - 同步通知所有订阅者"""
        handlers = self._handlers.get(event.event_type, [])
        for handler in handlers:
            try:
                await handler(event)
            except Exception as e:
                # 事件处理失败 - 在生产环境中应记录到死信队列
                print(f"[ERROR] 事件处理失败: {event.event_type} -> {e}")

# 全局事件总线实例
event_bus = EventBus()
```

```python
# order_service.py - 订单命令处理（写模型）
from event_bus import event_bus
from order_events import OrderCreated, OrderPaid

class OrderService:
    """订单服务 - 处理命令，产生事件
    
    核心原则：命令处理成功后发布事件，
    由事件订阅者异步执行后续操作（库存扣减、通知发送等）。
    """
    def __init__(self, order_store: dict):
        self.orders = order_store

    async def create_order(self, customer_id: str, items: list) -> str:
        """创建订单 - 产生 OrderCreated 事件"""
        order_id = f"ORD-{len(self.orders) + 1:06d}"
        total = sum(item["price"] * item["quantity"] for item in items)
        
        # 保存订单状态（写模型）
        self.orders[order_id] = {
            "id": order_id,
            "customer_id": customer_id,
            "items": items,
            "total": total,
            "status": "created",
        }
        
        # 发布事件 - 所有后续操作通过事件触发
        await event_bus.publish(OrderCreated(
            order_id=order_id,
            customer_id=customer_id,
            items=items,
            total=total,
        ))
        return order_id

    async def pay_order(self, order_id: str, payment_method: str) -> bool:
        """支付订单 - 产生 OrderPaid 事件"""
        order = self.orders.get(order_id)
        if not order or order["status"] != "created":
            return False
        
        order["status"] = "paid"
        await event_bus.publish(OrderPaid(
            order_id=order_id,
            payment_method=payment_method,
            amount=order["total"],
        ))
        return True
```

```python
        order_id = event.payload["order_id"]

        for item in items:
            sku = item["sku"]
            qty = item["quantity"]
            self.inventory[sku] = self.inventory.get(sku, 100) - qty
            
            if self.inventory[sku] < 0:
                # 库存不足 - 发布补偿事件
                print(f"[WARN] 库存不足: {sku}，需要触发补偿流程")
        
        print(f"[库存] 订单 {order_id} 库存扣减完成，剩余: {self.inventory}")

class NotificationHandler:
    """通知处理 - 订阅多个事件"""
    
    async def handle_order_created(self, event: Event):
        customer_id = event.payload["customer_id"]
        order_id = event.payload["order_id"]
        print(f"[通知] 向客户 {customer_id} 发送订单确认: {order_id}")

    async def handle_order_paid(self, event: Event):
        order_id = event.payload["order_id"]
        print(f"[通知] 向客户发送支付成功通知: {order_id}")

class AnalyticsHandler:
    """数据分析 - 订阅所有事件，构建读模型"""
    
    def __init__(self):
        self.revenue = 0.0
        self.order_count = 0

    async def handle_order_paid(self, event: Event):
        amount = event.payload["amount"]
        self.revenue += amount
        self.order_count += 1
        print(f"[分析] 累计收入: ¥{self.revenue:.2f}，订单数: {self.order_count}")
```

### 6.3 架构评审（35分钟）

使用 ATAM（Architecture Tradeoff Analysis Method）方法的关键步骤，评审上述架构方案：

| 同步调用为主 | ... | ... | ... | ... |

**Step 2：生成质量属性场景（10分钟）**

提出至少5个具体的、可测试的质量属性场景：

质量属性场景格式：

期望 [响应] 为 [响应度量]

示例：
[万人考生] 在 [考试开始时刻]，对 [考试服务] 施加 [同时提交答案]，
期望 [所有答案保存成功] 为 [零丢失且延迟 < 1秒]

**Step 3：分析架构方法（15分钟）**

对每个"高优先级 + 高难度"的场景，深入分析架构方案的应对能力：

场景：万人同时在线考试

架构方案的应对分析：
- 共享数据库瓶颈：MySQL 单机写入 TPS 约 3000-5000，
  万人同时提交会超出极限
  → 风险等级：高
  → 建议：考试服务独立数据库 + 写入缓冲

- 同步调用阻塞：如果考试提交需要同步调用支付服务验证付费状态，
  支付服务故障会导致考试提交失败
  → 风险等级：高
  → 建议：引入断路器 + 本地缓存付费状态

- 共享 Redis 连接池：高并发下 Redis 连接池可能耗尽，
  影响所有服务
  → 风险等级：中
  → 建议：每个服务独立的 Redis 连接池

### 进阶建议

1. **阅读经典文献**：Grady Booch《Object-Oriented Analysis and Design with Applications》、Martin Fowler《Patterns of Enterprise Application Architecture》、Sam Newman《Building Microservices》

2. **参与开源项目**：找一个你感兴趣的开源项目，分析其架构风格选择，尝试提出架构改进建议

3. **建立个人架构决策库**：记录你在实际项目中做出的架构决策，包括背景、备选方案、选择理由和实际效果，形成自己的"架构经验数据库"

4. **实践架构适应度函数**：在项目中引入 ArchUnit（Java）、Deptry（Python）等工具，将架构约束编码为自动化检查

5. **定期架构评审**：每季度对项目进行一次架构健康度评估，及早发现和治理架构腐化
