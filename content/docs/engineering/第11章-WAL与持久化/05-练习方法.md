## 练习1：实现简化版WAL引擎

### 为什么要做这个练习

WAL的理论在前面的章节中已经反复出现：ARIES的三阶段恢复（分析→重做→撤销）、LSN的全局有序性、日志先于数据的WAL规则。但理论和实现之间往往隔着一道鸿沟——你真的理解"prev_lsn如何形成反向链表"吗？"redo阶段为什么要检查page_lsn"？"undo为什么要沿着prev_lsn回溯"？

这个练习要求你从零构建一个支持崩溃恢复的简化WAL引擎。你会亲手实现日志记录的序列化、LSN的递增分配、ARIES三阶段恢复的完整流程。完成后，你将对WAL的每一个字段、每一个阶段有肌肉记忆级的理解。

### 理论回顾

在动手之前，快速回顾ARIES恢复的三个阶段：

1. **分析阶段（Analysis）**：从日志起点扫描到末尾，确定崩溃时的活跃事务集合和需要重做的最早LSN（redo_lsn）
2. **重做阶段（Redo）**：从redo_lsn开始重放所有日志，将数据页面恢复到崩溃时刻的状态（无论事务是否已提交）
3. **撤销阶段（Undo）**：对分析阶段识别出的未提交事务，沿着prev_lsn反向链逐条回滚

关键数据结构：
- **LSN（日志序列号）**：全局单调递增，唯一标识每条日志记录
- **prev_lsn**：同一事务中前一条日志的LSN，形成事务内的反向链表
- **page_lsn**：每个数据页面上记录的最后应用的LSN，用于重做时的幂等性判断

### 前置条件

- Python 3.8+
- 基本的数据结构知识（链表、哈希表）
- 理解二进制文件读写和struct模块

### 实现步骤

#### Step 1：定义日志记录格式

日志记录是WAL系统的基本单元。每条记录包含操作的完整描述，使得恢复时能够精确重做或撤销。注意before_image和after_image的设计——它们分别用于undo和redo。

```python
import struct
import os
import json
import time
from dataclasses import dataclass, field
from typing import Optional, List, Dict

@dataclass
class LogRecord:
    """WAL日志记录
    
    字段说明：
    - lsn: 日志序列号，全局单调递增，唯一标识此条记录
    - prev_lsn: 同一事务的前一条日志LSN，形成反向链表用于undo
    - txn_id: 事务ID，标识此记录属于哪个事务
    - op_type: 操作类型（INSERT/UPDATE/DELETE/COMMIT/ABORT）
    - page_id: 被修改的数据页面ID
    - offset: 页面内修改的起始偏移量
    - before_image: 修改前的数据（用于undo回滚）
    - after_image: 修改后的数据（用于redo重做）
    """
    lsn: int = 0
    prev_lsn: int = 0
    txn_id: int = 0
    op_type: str = ''
    page_id: int = 0
    offset: int = 0
    before_image: bytes = b''
    after_image: bytes = b''
    
    def serialize(self) -> bytes:
        """序列化为字节流
        
        格式: [4字节长度][JSON数据]
        长度前缀使得反序列化时能正确定位下一条记录的起始位置。
        """
        data = json.dumps({
            'lsn': self.lsn,
            'prev_lsn': self.prev_lsn,
            'txn_id': self.txn_id,
            'op_type': self.op_type,
            'page_id': self.page_id,
            'offset': self.offset,
            'before': self.before_image.hex(),
            'after': self.after_image.hex()
        }).encode()
        return struct.pack('I', len(data)) + data
    
    @classmethod
    def deserialize(cls, data: bytes) -> 'LogRecord':
        """从字节流反序列化"""
        length = struct.unpack('I', data[:4])[0]
        obj = json.loads(data[4:4+length])
        return cls(
            lsn=obj['lsn'],
            prev_lsn=obj['prev_lsn'],
            txn_id=obj['txn_id'],
            op_type=obj['op_type'],
            page_id=obj['page_id'],
            offset=obj['offset'],
            before_image=bytes.fromhex(obj['before']),
            after_image=bytes.fromhex(obj['after'])
        )
    
    def __str__(self) -> str:
        return (f"LSN={self.lsn} TXN={self.txn_id} "
                f"OP={self.op_type} PAGE={self.page_id} "
                f"PREV_LSN={self.prev_lsn}")
```

#### Step 2：实现日志管理器

日志管理器负责LSN的分配、日志记录的缓冲写入和刷盘。关键点：LSN的分配必须是原子的递增操作，刷盘时必须调用fsync确保持久化。

```python
class SimpleWALManager:
    """简化版WAL管理器
    
    核心职责：
    1. LSN分配：全局单调递增，保证日志的有序性
    2. 日志写入：先写入内存缓冲区，批量刷盘提高效率
    3. 日志扫描：支持从指定LSN开始顺序扫描
    """
    
    def __init__(self, log_file: str):
        self.log_file = log_file
        self.lsn_counter = 0
        self.buffer = bytearray()
        # 记录每个事务的最后LSN，用于快速定位
        self.txn_last_lsn: Dict[int, int] = {}
        self._load_last_lsn()
    
    def _load_last_lsn(self):
        """从现有日志文件恢复LSN计数器
        
        崩溃恢复时，WAL管理器需要从磁盘上的日志文件中
        找到最后一条记录的LSN，确保新分配的LSN不会冲突。
        """
        if not os.path.exists(self.log_file):
            return
        with open(self.log_file, 'rb') as f:
            data = f.read()
        pos = 0
        while pos < len(data):
            if pos + 4 > len(data):
                break  # 文件截断，日志不完整
            length = struct.unpack('I', data[pos:pos+4])[0]
            if pos + 4 + length > len(data):
                break  # 最后一条记录不完整，丢弃
            try:
                record = LogRecord.deserialize(data[pos:])
                self.lsn_counter = record.lsn + 1
                if record.txn_id not in (0, -1):
                    self.txn_last_lsn[record.txn_id] = record.lsn
                pos += 4 + length
            except (json.JSONDecodeError, struct.error):
                break  # 解析失败，停止扫描
    
    def append_log(self, record: LogRecord) -> int:
        """追加日志记录到缓冲区
        
        注意：此时只写入内存缓冲区，尚未持久化到磁盘。
        真正的持久化在flush()中通过fsync完成。
        返回值是分配给此记录的LSN。
        """
        record.lsn = self.lsn_counter
        self.lsn_counter += 1
        
        serialized = record.serialize()
        self.buffer.extend(serialized)
        
        # 维护事务到LSN的映射
        if record.op_type == 'COMMIT':
            self.txn_last_lsn.pop(record.txn_id, None)
        elif record.op_type not in ('ABORT',):
            self.txn_last_lsn[record.txn_id] = record.lsn
        
        return record.lsn
    
    def flush(self):
        """将缓冲区的日志刷盘
        
        这是WAL规则的关键执行点：日志必须在数据修改生效前持久化。
        fsync确保数据真正写入磁盘而非停留在操作系统页缓存中。
        """
        if not self.buffer:
            return
        with open(self.log_file, 'ab') as f:
            f.write(self.buffer)
            f.flush()
            os.fsync(f.fileno())
        self.buffer.clear()
    
    def scan_from(self, start_lsn: int) -> List[LogRecord]:
        """从指定LSN开始扫描日志文件
        
        用于恢复时的日志重放。返回所有LSN >= start_lsn的记录。
        """
        records = []
        if not os.path.exists(self.log_file):
            return records
        with open(self.log_file, 'rb') as f:
            data = f.read()
        pos = 0
        while pos < len(data):
            if pos + 4 > len(data):
                break
            length = struct.unpack('I', data[pos:pos+4])[0]
            if pos + 4 + length > len(data):
                break
            try:
                record = LogRecord.deserialize(data[pos:])
                if record.lsn >= start_lsn:
                    records.append(record)
                pos += 4 + length
            except (json.JSONDecodeError, struct.error):
                break
        return records
    
    def scan_all(self) -> List[LogRecord]:
        """扫描全部日志"""
        return self.scan_from(0)
```

#### Step 3：实现数据页面和缓冲池

缓冲池是WAL系统中日志与数据文件之间的桥梁。每个页面维护自己的page_lsn，这是重做阶段判断"此日志是否已经应用过"的关键依据。

```python
PAGE_SIZE = 4096

class DataPage:
    """数据页面
    
    每个页面维护page_lsn，记录最后应用到此页面的日志LSN。
    重做阶段通过比较日志LSN和page_lsn来决定是否需要重放。
    如果 page_lsn >= rec.lsn，说明此操作已经应用过，跳过（幂等性保证）。
    """
    def __init__(self, page_id: int):
        self.page_id = page_id
        self.data = bytearray(PAGE_SIZE)
        self.page_lsn = 0
        self.is_dirty = False
    
    def apply(self, offset: int, value: bytes):
        """应用修改（可能是redo的after_image，也可能是undo的before_image）"""
        self.data[offset:offset+len(value)] = value
        self.is_dirty = True
    
    def __str__(self) -> str:
        preview = bytes(self.data[:16]).hex()
        return f"Page(id={self.page_id}, lsn={self.page_lsn}, dirty={self.is_dirty}, data={preview}...)"


class SimpleBufferPool:
    """简化版缓冲池
    
    真实的缓冲池使用LRU等淘汰策略管理固定容量的页面缓存。
    这里简化为按需加载，重点演示WAL恢复流程。
    """
    def __init__(self, capacity: int = 100):
        self.capacity = capacity
        self.pages: Dict[int, DataPage] = {}
    
    def fetch(self, page_id: int) -> DataPage:
        """获取页面，不存在则创建（简化版：从空页面开始）"""
        if page_id not in self.pages:
            self.pages[page_id] = DataPage(page_id)
        return self.pages[page_id]
    
    def mark_dirty(self, page_id: int, lsn: int):
        """标记页面为脏页"""
        page = self.fetch(page_id)
        page.is_dirty = True
        if lsn > page.page_lsn:
            page.page_lsn = lsn
```

#### Step 4：实现ARIES恢复管理器

这是整个练习的核心。三阶段恢复流程的每一步都有明确的设计意图：

- **分析阶段**：扫描全部日志，建立"哪些事务未完成"和"哪些页面被修改"的全局视图
- **重做阶段**：从最早的脏页LSN开始，将所有页面恢复到崩溃时刻的状态（包括未提交事务的修改）
- **撤销阶段**：沿着每条未提交事务的prev_lsn链，反向逐条回滚

```python
class RecoveryManager:
    """ARIES恢复管理器
    
    实现标准的三阶段恢复：
    1. Analysis: 确定活跃事务和重做起始点
    2. Redo: 重放所有日志，恢复崩溃前状态
    3. Undo: 回滚未提交事务
    """
    
    def __init__(self, wal_manager: SimpleWALManager, 
                 buffer_pool: SimpleBufferPool):
        self.wal = wal_manager
        self.bp = buffer_pool
    
    def recover(self):
        """执行完整的ARIES恢复流程"""
        print("=" * 50)
        print("ARIES Recovery: 开始恢复")
        print("=" * 50)
        
        # 阶段1：分析
        active_txns, dirty_pages, redo_start = self._analysis_phase()
        print(f"\n[Analysis] 活跃事务: {len(active_txns)}个 "
              f"{list(active_txns.keys())}")
        print(f"[Analysis] 脏页面: {len(dirty_pages)}个 "
              f"{list(dirty_pages.keys())}")
        print(f"[Analysis] Redo起始LSN: {redo_start}")
        
        # 阶段2：重做
        redo_count = self._redo_phase(redo_start, dirty_pages)
        print(f"\n[Redo] 重放了 {redo_count} 条日志记录")
        
        # 阶段3：撤销
        undo_count = self._undo_phase(active_txns)
        print(f"\n[Undo] 回滚了 {undo_count} 条日志记录")
        
        print("\n" + "=" * 50)
        print("ARIES Recovery: 恢复完成")
        print("=" * 50)
        
        return active_txns, dirty_pages
    
    def _analysis_phase(self):
        """分析阶段：扫描全部日志，建立恢复所需的全局视图
        
        遍历所有日志记录，记录：
        - 每个事务的最后LSN（用于Undo时的起点）
        - 每个被修改页面的最小LSN（用于Redo的起始点）
        - 哪些事务已提交（COMMIT）或中止（ABORT）
        """
        records = self.wal.scan_all()
        active_txns = {}       # txn_id -> last_lsn
        dirty_pages = {}       # page_id -> min_lsn
        redo_start = float('inf')
        
        for rec in records:
            if rec.op_type in ('INSERT', 'UPDATE', 'DELETE'):
                # 数据操作：事务仍然活跃，记录该页面被修改
                active_txns[rec.txn_id] = rec.lsn
                if rec.page_id not in dirty_pages:
                    dirty_pages[rec.page_id] = rec.lsn
                redo_start = min(redo_start, rec.lsn)
            elif rec.op_type == 'COMMIT':
                # 已提交，从活跃事务列表移除
                active_txns.pop(rec.txn_id, None)
            elif rec.op_type == 'ABORT':
                # 已中止但仍需Undo（如果已经写入了数据页面）
                # 保留活跃状态以便Undo阶段处理
                active_txns[rec.txn_id] = rec.lsn
        
        if redo_start == float('inf'):
            redo_start = 0
        
        return active_txns, dirty_pages, redo_start
    
    def _redo_phase(self, redo_start: int, dirty_pages: Dict[int, int]):
        """重做阶段：从redo_start开始重放所有日志
        
        关键逻辑：对于每条日志记录，检查 page_lsn < rec.lsn。
        如果 page_lsn >= rec.lsn，说明此操作已经应用到页面上（崩溃前就写入了），
        无需重复应用。这就是幂等性保证。
        
        注意：重做阶段会重放已提交和未提交事务的所有操作。
        未提交事务的回滚在Undo阶段处理。
        """
        records = self.wal.scan_from(redo_start)
        redo_count = 0
        
        for rec in records:
            if rec.op_type not in ('INSERT', 'UPDATE', 'DELETE'):
                continue
            if rec.page_id not in dirty_pages:
                continue
            
            page = self.bp.fetch(rec.page_id)
            # 幂等性检查：只有当此日志尚未应用时才重放
            if page.page_lsn < rec.lsn:
                page.apply(rec.offset, rec.after_image)
                page.page_lsn = rec.lsn
                redo_count += 1
                print(f"  Redo: {rec}")
        
        return redo_count
    
    def _undo_phase(self, active_txns: Dict[int, int]):
        """撤销阶段：回滚所有未提交事务
        
        对每个活跃（未提交）事务，从其最后一条日志开始，
        沿着prev_lsn反向链逐条回滚。
        
        关键设计：prev_lsn形成事务内的单向链表，
        使得Undo可以精确地只回滚特定事务的操作，
        而不影响已提交事务的数据。
        """
        undo_count = 0
        
        for txn_id, last_lsn in active_txns.items():
            print(f"\n  Undo事务 {txn_id}，起始LSN={last_lsn}")
            current_lsn = last_lsn
            
            while current_lsn != 0:
                # 通过LSN定位具体记录
                record = self._find_record(current_lsn)
                if record is None:
                    print(f"    警告: 无法找到LSN={current_lsn}的记录")
                    break
                
                if record.op_type in ('INSERT', 'UPDATE', 'DELETE'):
                    # 使用before_image回滚
                    page = self.bp.fetch(record.page_id)
                    page.apply(record.offset, record.before_image)
                    print(f"    回滚: {record}")
                    undo_count += 1
                
                # 沿着反向链回溯
                current_lsn = record.prev_lsn
        
        return undo_count
    
    def _find_record(self, target_lsn: int) -> Optional[LogRecord]:
        """通过LSN查找具体的日志记录"""
        for rec in self.wal.scan_from(target_lsn):
            if rec.lsn == target_lsn:
                return rec
            if rec.lsn > target_lsn:
                break
        return None
```

#### Step 5：测试崩溃恢复

现在用三个精心设计的场景来验证恢复逻辑。每个场景模拟一种典型的崩溃模式。

```python
def test_crash_recovery():
    """测试崩溃恢复 —— 三种典型场景
    
    场景1: 正常提交 → 恢复后数据应保留
    场景2: 未提交事务 → 恢复后应回滚
    场景3: 混合场景 → 部分提交部分回滚
    """
    import shutil
    
    test_dir = '/tmp/wal_test'
    if os.path.exists(test_dir):
        shutil.rmtree(test_dir)
    os.makedirs(test_dir)
    
    # ========== 场景1: 正常提交 ==========
    print("\n" + "=" * 50)
    print("场景1: 正常提交 —— 事务完整执行并提交")
    print("=" * 50)
    
    wal1 = SimpleWALManager(os.path.join(test_dir, 'wal1.log'))
    bp1 = SimpleBufferPool()
    
    # 事务1: 修改页面1，偏移0处写入8字节
    rec = LogRecord(0, 0, 1, 'UPDATE', 1, 0, 
                    b'\x00' * 8, b'\xAA' * 8)
    wal1.append_log(rec)
    
    # 事务1: 提交
    rec = LogRecord(0, 0, 1, 'COMMIT', 0, 0, b'', b'')
    wal1.append_log(rec)
    
    wal1.flush()
    print("写入: 事务1更新页面1并提交")
    
    # 模拟崩溃恢复
    wal1_recover = SimpleWALManager(os.path.join(test_dir, 'wal1.log'))
    bp1_recover = SimpleBufferPool()

    rm1.recover()
    
    page1 = bp1_recover.fetch(1)
    result = bytes(page1.data[:8])
    expected = b'\xAA' * 8
    status = "✓ 正确" if result == expected else "✗ 错误"
    print(f"\n验证: 页面1 = {result.hex()}, 期望 = {expected.hex()} → {status}")
    
    # ========== 场景2: 未提交事务（模拟崩溃）==========
    print("\n" + "=" * 50)
    print("场景2: 未提交事务 —— 事务修改了数据但未提交就崩溃")
    print("=" * 50)
    
    wal2 = SimpleWALManager(os.path.join(test_dir, 'wal2.log'))
    bp2 = SimpleBufferPool()
    
    # 事务2: 修改页面2
    rec = LogRecord(0, 0, 2, 'UPDATE', 2, 0, 
                    b'\x00' * 8, b'\xBB' * 8)
    wal2.append_log(rec)
    # 注意：没有COMMIT！模拟崩溃
```

## 练习2：分析PostgreSQL的WAL行为

### 为什么要做这个练习

练习1让你从零构建了WAL引擎，但它毕竟是简化版。PostgreSQL的WAL是一个经过30多年迭代的生产级实现，有着远比简化版复杂的日志格式、资源管理器体系和复制支持。通过pg_waldump工具直接观察真实的WAL记录，你能看到理论与工程实践之间的差距，理解为什么真实系统需要那么多"额外"的设计。

### 理论回顾

PostgreSQL的WAL记录按**资源管理器（Resource Manager）**分类：

| 资源管理器 | 缩写 | 负责的日志类型 |
|-----------|------|-------------|
| XLOG | XLOG | 数据库启动/关闭、检查点、切换 |
| Transaction | Xact | 事务提交/中止 |
| Heap | Heap | 表数据的INSERT/UPDATE/DELETE |
| Btree | Btree | B+树索引的操作 |
| SMGR | SMGR | 存储管理器（文件创建/扩展） |

每条WAL记录都包含：LSN（唯一标识）、Prev LSN（反向链表指针）、事务ID、CRC校验、以及特定于资源管理器的数据。

### 前置条件

- PostgreSQL 14+ 安装并运行
- pg_waldump工具可用（通常随PostgreSQL安装）
- 有权限创建测试数据库

## 练习3：fsync性能基准测试

### 为什么要做这个练习

在WAL的所有工程挑战中，fsync是最关键的性能瓶颈。一次fsync在HDD上可能需要10-20ms，在SSD上也需要0.1-1ms。组提交（Group Commit）之所以能将吞吐量提升10-100倍，本质上就是通过合并多个事务的fsync来摊销这个固定开销。

这个练习让你亲手测量fsync的真实性能，理解数据大小、存储介质、文件系统对fsync延迟的影响。这些数据将帮助你直观理解为什么WAL设计中"减少fsync次数"是第一优先级的优化目标。

### 理论回顾

**fsync vs fdatasync：**

| 系统调用 | 保证 | 开销 |
|---------|------|------|
| `fsync(fd)` | 文件数据+元数据（inode、mtime等）都持久化 | 较大 |
| `fdatasync(fd)` | 文件数据持久化，元数据仅在必要时持久化（如文件大小变化） | 较小 |

对于WAL日志文件来说，文件只做追加写入（大小持续增长），fdatasync就够了——元数据中的文件大小变化会在fdatasync中被持久化。但如果是就地更新的文件（如数据页），元数据不一定变化，fdatasync可以跳过元数据刷盘。

**关键认知**：fsync的延迟主要来自磁盘硬件的写入延迟，与写入数据量关系不大。写4KB和写1MB的fsync延迟差异通常不超过2倍，因为瓶颈在磁盘的写入确认延迟而非数据传输。

### 前置条件

- Linux环境
- Python 3.8+
- 需要root权限或`/proc/sys/vm/drop_caches`的写入权限（用于清除页缓存）

### 操作步骤

#### Step 1：基础fsync基准测试

```python
import os
import time
import statistics
import json

def benchmark_single_fsync(filepath, data_size, use_fdatasync=False):
    """单次写入+fsync的延迟测试
    
    测量单次 write+fsync 的端到端延迟。
    这是WAL中每次提交的最小I/O开销。
    """
    func_name = 'fdatasync' if use_fdatasync else 'fsync'
    sync_func = os.fdatasync if use_fdatasync else os.fsync
    
    results = []
    data = os.urandom(data_size)  # 用随机数据避免压缩优化
    
    for _ in range(200):  # 200次采样
        fd = os.open(filepath, os.O_WRONLY | os.O_CREAT | os.O_TRUNC, 0o644)
        
        start = time.perf_counter_ns()
        os.write(fd, data)
        sync_func(fd)
        elapsed_ns = time.perf_counter_ns() - start
        
        results.append(elapsed_ns / 1_000_000)  # 转换为毫秒
        os.close(fd)
    
    os.remove(filepath)
    results.sort()
    return {
        'func': func_name,
        'data_size': data_size,
        'p50': results[len(results)//2],
        'p95': results[int(len(results)*0.95)],
        'p99': results[int(len(results)*0.99)],
        'mean': statistics.mean(results),
        'min': results[0],
        'max': results[-1],
    }


def benchmark_batch_fsync(filepath, num_writes, write_size, batch_size):
    """批量写入+合并fsync的延迟测试
    
    模拟组提交：将多次写入合并为一次fsync。
    batch_size=1 等价于每次写入都fsync（无组提交）
    batch_size=N 等价于N次写入后才fsync（组提交）
    """
    results = []
    data = os.urandom(write_size)
    
    for trial in range(100):
        fd = os.open(filepath, os.O_WRONLY | os.O_CREAT | os.O_TRUNC, 0o644)
        
        start = time.perf_counter_ns()
        for i in range(num_writes):
            os.write(fd, data)
            if (i + 1) % batch_size == 0:
                os.fsync(fd)
        # 确保最后一次不足batch_size的也刷盘
        if num_writes % batch_size != 0:
            os.fsync(fd)
        elapsed_ns = time.perf_counter_ns() - start
        
        results.append(elapsed_ns / 1_000_000)
        os.close(fd)
    
    os.remove(filepath)
    results.sort()
    total_ops = (num_writes + batch_size - 1) // batch_size  # fsync次数
    return {
        'batch_size': batch_size,
        'total_writes': num_writes,
        'fsync_count': total_ops,
        'total_time_ms': statistics.mean(results),
        'per_fsync_ms': statistics.mean(results) / total_ops,
        'throughput_ops_sec': num_writes / (statistics.mean(results) / 1000),
    }
```

## 练习4：SQLite WAL模式对比测试

### 为什么要做这个练习

SQLite是最广泛部署的数据库引擎，它的WAL模式是一个优雅的简化设计。与PostgreSQL/MySQL的复杂WAL系统不同，SQLite的WAL只有三个文件（main DB、WAL file、shared-memory file），但依然实现了崩溃恢复和读写并发。通过对比delete journal模式和WAL模式的行为差异，你能深入理解"WAL模式如何支持并发读写"以及"为什么WAL模式下读操作不会被写操作阻塞"。

### 理论回顾

**SQLite的两种日志模式：**

| 特性 | DELETE模式（默认） | WAL模式 |
|------|------------------|---------|
| 日志机制 | 修改前将原始页面写入journal文件 | 修改后将新页面追加到WAL文件 |
| 并发读写 | 写操作阻塞所有读操作 | 读写互不阻塞 |
| 提交开销 | 每次提交需删除journal文件 | 追加写入，无需删除 |
| 检查点 | 无显式检查点 | 定期将WAL内容合并回主数据库 |
| 适用场景 | 写少读多、单连接 | 读写并发、多连接 |

### 前置条件

- Python 3.8+
- SQLite3（Python内置，无需额外安装）
- Linux/macOS/Windows均可

```python
import sqlite3
import time
import os

def benchmark_sequential_writes(mode, num_rows=50000, batch_size=1000):
    """顺序写入性能测试
    
    测试大量INSERT操作在不同日志模式下的性能差异。
    batch_size控制每次事务包含的INSERT数量。
    """
    db_path = f'/tmp/wal_bench_{mode}.db'
    if os.path.exists(db_path):
        os.remove(db_path)
    for suffix in ('-wal', '-shm'):
        if os.path.exists(db_path + suffix):
            os.remove(db_path + suffix)
    
    conn = sqlite3.connect(db_path)
    conn.execute(f'PRAGMA journal_mode={mode}')
    conn.execute('PRAGMA synchronous=NORMAL')  # 平衡安全与性能
    conn.execute('''CREATE TABLE data (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        key TEXT NOT NULL,
        value TEXT NOT NULL,
        created_at REAL
    )''')
    conn.commit()
    
    # 批量插入
    start = time.perf_counter()
    for batch_start in range(0, num_rows, batch_size):
        rows = [(f'key_{i}', f'value_{i}' * 10, time.time()) 
                for i in range(batch_start, min(batch_start + batch_size, num_rows))]
        conn.executemany('INSERT INTO data (key, value, created_at) VALUES (?, ?, ?)', rows)
        conn.commit()
    
    elapsed = time.perf_counter() - start
    conn.close()
    
    db_size = os.path.getsize(db_path)
    for suffix in ('-wal', '-shm'):
        p = db_path + suffix
        if os.path.exists(p):
            db_size += os.path.getsize(p)
    
    return {
        'mode': mode,
        'rows': num_rows,
        'elapsed': elapsed,
        'throughput': num_rows / elapsed,
        'db_size_mb': db_size / (1024 * 1024),
    }
```

#### Step 2：读写并发测试

```python
import threading
import random

def test_concurrent_read_write(mode, num_readers=4, duration_seconds=5):
    """读写并发测试
    
    多个读线程同时执行SELECT查询，一个写线程持续执行UPDATE。
    测试两种模式下的并发行为差异。
    
    DELETE模式：写操作会获取EXCLUSIVE锁，读操作必须等待
    WAL模式：读操作使用snapshot isolation，不受写操作影响
    """
    db_path = f'/tmp/wal_concurrent_{mode}.db'
    if os.path.exists(db_path):
        os.remove(db_path)
    for suffix in ('-wal', '-shm'):
        p = db_path + suffix
        if os.path.exists(p):
            os.remove(p)
    
    # 初始化数据
    conn = sqlite3.connect(db_path)
    conn.execute(f'PRAGMA journal_mode={mode}')
    conn.execute('''CREATE TABLE accounts (
        id INTEGER PRIMARY KEY,
        name TEXT,
        balance REAL
    )''')
    conn.executemany('INSERT INTO accounts VALUES (?, ?, ?)',
                     [(i, f'User_{i}', 1000.0) for i in range(1000)])
    conn.commit()
    conn.close()
    
    stop_flag = threading.Event()
    results = {
        'reads': 0, 'read_errors': 0, 'read_times': [],
        'writes': 0, 'write_errors': 0, 'write_times': [],
    }
    lock = threading.Lock()
    
    def reader_task(reader_id):
        """读线程：持续执行随机SELECT查询"""
        rconn = sqlite3.connect(db_path, timeout=5)
        rconn.execute(f'PRAGMA journal_mode={mode}')
        
        while not stop_flag.is_set():
            try:
                uid = random.randint(1, 1000)
                start = time.perf_counter_ns()
                rconn.execute(
                    'SELECT * FROM accounts WHERE id = ?', (uid,)
                ).fetchone()
                elapsed = (time.perf_counter_ns() - start) / 1_000_000
                
                with lock:
                    results['reads'] += 1
                    results['read_times'].append(elapsed)
            except sqlite3.OperationalError as e:
                with lock:
                    results['read_errors'] += 1
        
        rconn.close()
    
    def writer_task():
        """写线程：持续执行UPDATE操作"""
        wconn = sqlite3.connect(db_path, timeout=5)
        wconn.execute(f'PRAGMA journal_mode={mode}')
        
        counter = 0
        while not stop_flag.is_set():
            try:
                uid = random.randint(1, 1000)
                start = time.perf_counter_ns()
                wconn.execute(
                    'UPDATE accounts SET balance = balance + 1 WHERE id = ?',
                    (uid,)
                )
                wconn.commit()
                elapsed = (time.perf_counter_ns() - start) / 1_000_000
                
                with lock:
                    results['writes'] += 1
                    results['write_times'].append(elapsed)
                counter += 1
            except sqlite3.OperationalError as e:
                with lock:
                    results['write_errors'] += 1
        
        wconn.close()
    
    # 启动线程
    threads = []
    for i in range(num_readers):
        t = threading.Thread(target=reader_task, args=(i,))
        threads.append(t)
        t.start()
    
    writer = threading.Thread(target=writer_task)
    writer.start()
    
    # 运行指定时间
    time.sleep(duration_seconds)
    stop_flag.set()
    
    for t in threads:
        t.join(timeout=5)
    writer.join(timeout=5)
    
    # 关闭数据库
    if os.path.exists(db_path):
        os.remove(db_path)
    for suffix in ('-wal', '-shm'):
        p = db_path + suffix
        if os.path.exists(p):
            os.remove(p)
    
    return results
```
