## 2. 算法与数据结构优化

算法复杂度是性能的天花板。选择正确的数据结构，往往比任何微优化都更有效。

### 2.1 复杂度感知的代码重写

**案例：从 O(n²) 到 O(n)**

```python
# 反面: O(n²) — 嵌套循环查找重复
def find_duplicates_slow(data):
    duplicates = []
    for i in range(len(data)):
        for j in range(i + 1, len(data)):  # O(n²)
            if data[i] == data[j] and data[i] not in duplicates:
                duplicates.append(data[i])
    return duplicates

# 正面: O(n) — 哈希集合查找重复
def find_duplicates_fast(data):
    seen = set()
    duplicates = set()
    for item in data:                        # O(n)
        if item in seen:                     # O(1)
            duplicates.add(item)
        else:
            seen.add(item)
    return list(duplicates)

# 性能差异: n=10000 时, O(n²) ≈ 50s, O(n) ≈ 0.003s
```

**案例：避免不必要的排序**

```python
# 反面: 排序后取前 K 个 — O(n log n)
def top_k_slow(data, k):
    return sorted(data, reverse=True)[:k]

# 正面: 堆排序取前 K 个 — O(n log k)
import heapq
def top_k_fast(data, k):
    return heapq.nlargest(k, data)  # 使用最小堆，维护大小为 k 的堆

# 当 k << n 时（如 n=100万, k=10），heapq 比 sorted 快 50 倍以上
```

### 3.1 减少内存分配

```python
# 反面: 字符串拼接创建大量临时对象
def build_log_slow(entries):
    result = ""
    for entry in entries:
        result += f"[{entry['time']}] {entry['msg']}\n"  # 每次 += 创建新字符串
    return result

# 正面: 使用 io.StringIO 缓冲
import io
def build_log_fast(entries):
    buf = io.StringIO()
    for entry in entries:
        buf.write(f"[{entry['time']}] {entry['msg']}\n")  # 原地写入
    return buf.getvalue()

# 对于 10 万条日志: slow ~200ms + 800MB 分配, fast ~15ms + 5MB 分配
```

```go
// Go: 使用 sync.Pool 复用对象
var bufPool = sync.Pool{
    New: func() interface{} {
        return new(bytes.Buffer)
    },
}

func processRequest(data []byte) string {
    buf := bufPool.Get().(*bytes.Buffer)
    buf.Reset()
    defer bufPool.Put(buf)

    buf.Write(data)
    return buf.String()
}

// 效果: 在高并发场景下，减少 GC 压力 50-80%
```

### 3.2 零拷贝与内存映射

```python
# 反面: 读取大文件占用双倍内存
def process_file_slow(path):
    with open(path, 'rb') as f:
        data = f.read()  # 整个文件加载到内存
    return analyze(data)

# 正面: 内存映射，按需加载
import mmap
def process_file_fast(path):

        mm.close()

```

## 4. I/O 优化技巧

### 4.1 批量处理与缓冲

```python
# 反面: 逐条数据库写入 — N 次网络往返
def save_records_slow(db, records):
    for record in records:                    # 10000 条记录
        db.execute("INSERT INTO logs VALUES (?, ?, ?)",
                   (record['ts'], record['msg'], record['level']))
    db.commit()                               # 1 次提交，但 10000 次网络往返

# 正面: 批量写入 — 1 次网络往返
def save_records_fast(db, records):
    db.executemany("INSERT INTO logs VALUES (?, ?, ?)",
                   [(r['ts'], r['msg'], r['level']) for r in records])
    db.commit()                               # 1 次提交 + 1 次网络往返

# 正面: 分块批量（避免单次事务过大）
def save_records_chunked(db, records, chunk_size=1000):
    for i in range(0, len(records), chunk_size):
        chunk = records[i:i + chunk_size]
        db.executemany("INSERT INTO logs VALUES (?, ?, ?)",
                       [(r['ts'], r['msg'], r['level']) for r in chunk])
        db.commit()
    # 效果: 10000 条记录从 ~30s 降到 ~0.5s
```

## 5. 并发优化技巧

### 5.1 锁粒度优化

```go
// 反面: 全局锁 — 所有操作互斥
type GlobalCache struct {
    mu    sync.RWMutex
    items map[string]*Item
}

func (c *GlobalCache) Get(key string) *Item {
    c.mu.RLock()          // 所有读操作共享这把锁
    defer c.mu.RUnlock()  // 100 个 goroutine 排队等锁
    return c.items[key]
}

// 正面: 分段锁 — 按 key 哈希分散到不同锁
type ShardedCache struct {
    shards [256]struct {
        mu    sync.RWMutex
        items map[string]*Item
    }
}

func (c *ShardedCache) Get(key string) *Item {
    shard := &amp;c.shards[hash(key)&amp;255]  // 256 把锁，冲突概率降低 256 倍
    shard.mu.RLock()
    defer shard.mu.RUnlock()
    return shard.items[key]
}

// 实测: 100 并发读写时，分段锁吞吐量是全局锁的 10-50 倍
```

### 5.2 无锁编程

```go
// 原子操作替代简单锁
var (
    hitCount  int64
    missCount int64
)

func recordHit() {
    atomic.AddInt64(&amp;hitCount, 1)  // 无锁 CAS，纳秒级
}

// CAS 自旋实现无锁队列
func pushIfAvailable(q *Queue, item *Item) bool {
    for {
        old := q.head.Load()
        item.next = old
        if q.head.CompareAndSwap(old, item) {  // CAS: 如果 head 没变就写入
            return true
        }
        // CAS 失败，说明其他 goroutine 修改了 head，重试
    }
}
```

### 5.3 goroutine/协程管理

```go
// 反面: 无限制创建 goroutine（内存爆炸）
func processAll(items []Item) {
    for _, item := range items {
        go process(item)  // 100万个 item = 100万个 goroutine，每个占 2-8KB 栈
    }
    // 总内存: 2GB - 8GB，极可能 OOM
}

// 正面: 使用 worker pool 控制并发度
func processAll(items []Item) {
    workers := 100              // 限制最大并发
    ch := make(chan Item, 1000) // 带缓冲的任务队列

    // 启动固定数量的 worker
    var wg sync.WaitGroup
    for i := 0; i < workers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for item := range ch {
                process(item)
            }
        }()
    }

    // 分发任务
    for _, item := range items {
        ch <- item
    }
    close(ch)
    wg.Wait()  // 等待所有 worker 完成
}
```

## 6. 数据库优化技巧

### 6.1 索引策略

```sql
-- 反面: 缺少索引，全表扫描
SELECT * FROM orders WHERE user_id = 123 AND status = 'paid';
-- EXPLAIN: type=ALL, rows=10000000 (全表扫描 1000万行)

-- 正面: 复合索引（列顺序很关键）
CREATE INDEX idx_orders_user_status ON orders(user_id, status);
-- EXPLAIN: type=ref, rows=5 (只扫描 5 行)

-- 关键原则: 索引列顺序 = WHERE 条件中等值查询的顺序
-- 范围查询的列放在最后:
CREATE INDEX idx_orders_user_time ON orders(user_id, created_at);
-- WHERE user_id=123 AND created_at > '2024-01-01'
-- → user_id 精确匹配, created_at 范围扫描

-- 覆盖索引: 避免回表
CREATE INDEX idx_orders_covering ON orders(user_id, status, amount);
-- SELECT amount FROM orders WHERE user_id=123 AND status='paid'
-- → 数据全在索引中，无需回表读取数据行
```

### 6.2 查询优化

```sql
-- 反面: SELECT * 传输多余数据
SELECT * FROM users WHERE age > 18;  -- 传输所有列，包括大字段

-- 正面: 只选需要的列
SELECT id, name, email FROM users WHERE age > 18;

-- 反面: 分页深度问题
SELECT * FROM orders ORDER BY id LIMIT 100000, 20;
-- 需要先扫描 100020 行再丢弃前 100000 行

-- 正面: 游标分页（基于上一页最后一条的 ID）
SELECT * FROM orders WHERE id > 100000 ORDER BY id LIMIT 20;
-- 直接从索引定位，无需扫描前 100000 行

-- 反面: 子查询导致性能差
SELECT * FROM users WHERE id IN (SELECT user_id FROM orders WHERE amount > 1000);

-- 正面: JOIN 通常更快
SELECT u.* FROM users u
INNER JOIN orders o ON u.id = o.user_id
WHERE o.amount > 1000;
```

### 6.3 连接池与事务管理

```python
# 反面: 长事务持有连接
def process_order(order_id):
    conn = get_connection()
    try:
        order = conn.query("SELECT * FROM orders WHERE id=?", order_id)
        # ... 大量业务逻辑（可能需要数秒）
        # 外部 API 调用、文件处理等

        conn.execute("UPDATE orders SET status=? WHERE id=?", (result, order_id))

        conn.release()                        # 连接被占用数秒

# 正面: 短事务 + 业务逻辑在事务外
def process_order(order_id):
    # 1. 事务外: 执行耗时操作
    conn = get_connection()
    order = conn.query("SELECT * FROM orders WHERE id=?", order_id)
    conn.release()

    result = external_api.call(order)         # 不占用数据库连接

    # 2. 快速事务: 只做数据更新
    conn = get_connection()
    try:
        conn.execute("UPDATE orders SET status=? WHERE id=?", (result, order_id))
        conn.commit()
    finally:
        conn.release()                        # 连接占用时间从数秒降到毫秒级
```

```python
orders = db.query("SELECT * FROM orders LIMIT 100")

user_map = {u.id: u for u in users}
```

```python
    resp = make_response(jsonify(products))

    # 强缓存: 浏览器直接使用本地缓存，不发请求
    resp.headers['Cache-Control'] = 'public, max-age=3600'  # 缓存 1 小时

    # 协商缓存: 浏览器带条件请求，服务端决定是否返回新数据
    resp.headers['ETag'] = hashlib.md5(resp.data).hexdigest()

    return resp

# 缓存层次:
# 浏览器缓存 → CDN 缓存 → 反向代理缓存(Nginx) → 应用缓存(Redis) → 数据库
# 每一层都减少请求到达数据库的概率
```


## 10. 优化决策框架

面对性能问题时，按以下顺序选择优化策略：

1. 算法优化  → 效果最大（O(n²)→O(n) 可达百倍提升）
       ↓
2. 架构优化  → 缓存、异步、分库分表（10-100倍）
       ↓
3. 并发优化  → 多线程、连接池、批量处理（2-10倍）
       ↓
4. 代码优化  → 减少分配、避免拷贝、内联（10%-50%）
       ↓
5. 系统优化  → 内核参数、编译器选项（5%-30%）

**不要跳步**。在算法 O(n²) 的情况下调 Nginx 参数毫无意义。先识别瓶颈所在层次，再选择对应层级的优化手段。

## 11. 常见误区与纠正

| 误区 | 为什么错 | 正确做法 |
|------|----------|----------|
| "加机器就能解决" | 扩容是兜底手段，不解决根本问题 | 先找瓶颈、做优化，再按需扩容 |
| "代码写得越快越好" | 过早微优化浪费时间，且可能降低可读性 | 先保证正确性和可读性，profiling 后针对性优化 |
| "缓存是万能药" | 缓存引入一致性问题、雪崩风险、内存消耗 | 评估缓存收益与成本，设计失效策略 |
| "优化所有代码" | 80%的耗时在20%的热点代码中 | 用 profiling 找到热点，只优化那 20% |
| "线上直接优化" | 无法控制影响范围，出问题难以回滚 | 在 staging 环境验证，灰度发布 |
| "优化一次就够了" | 业务增长、数据量变化会改变瓶颈位置 | 建立持续监控，定期性能回归测试 |
