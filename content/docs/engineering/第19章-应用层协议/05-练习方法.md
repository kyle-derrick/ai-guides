## 练习三：TLS 证书配置（预计 60 分钟）

**目标**：掌握 X.509 证书的结构、CA 信任链的验证过程，能独立配置自签名证书并搭建 HTTPS 服务。

### 步骤 1：解析真实证书（15 分钟）

```bash
# 查看任意 HTTPS 站点的证书详情
echo | openssl s_client -connect www.baidu.com:443 -servername www.baidu.com 2>/dev/null | \
  openssl x509 -noout -text

# 重点观察：
# 1. Issuer（签发者 CA）和 Subject（持有者）的区别
# 2. Validity（有效期）的 Not Before / Not After
# 3. Public Key Algorithm（公钥算法和密钥长度）
# 4. X509v3 extensions（SAN、Key Usage、Basic Constraints）
# 5. Signature Algorithm（签名算法）

# 对比 ECDSA 和 RSA 证书
echo | openssl s_client -connect cloudflare.com:443 2>/dev/null | \
  openssl x509 -noout -text | grep -E "Public Key Algorithm|Public-Key|Signature Algorithm"
```

### 步骤 2：自建 CA 和签发证书（20 分钟）

```bash
# 创建工作目录
mkdir -p ~/tls-practice &amp;&amp; cd ~/tls-practice

# 第一步：生成 CA 根证书
openssl genrsa -out ca.key 4096
openssl req -new -x509 -days 365 -key ca.key -out ca.crt \
  -subj "/C=CN/ST=Beijing/O=MyCA/CN=My Root CA"

# 第二步：生成服务器证书申请
openssl genrsa -out server.key 2048
openssl req -new -key server.key -out server.csr \
  -subj "/C=CN/ST=Beijing/O=MyServer/CN=localhost"

# 第三步：用 CA 签发服务器证书（带 SAN 扩展）
cat > server_ext.cnf << EOF
[v3_req]
basicConstraints = CA:FALSE
keyUsage = digitalSignature, keyEncipherment
extendedKeyUsage = serverAuth
subjectAltName = @alt_names

[alt_names]
DNS.1 = localhost
DNS.2 = *.example.com
IP.1 = 127.0.0.1
IP.2 = 192.168.1.100
EOF

openssl x509 -req -in server.csr -CA ca.crt -CAkey ca.key \
  -CAcreateserial -out server.crt -days 365 \
  -extfile server_ext.cnf -extensions v3_req

# 验证证书链
openssl verify -CAfile ca.crt server.crt
# 输出应为: server.crt: OK
```

### 步骤 3：搭建 HTTPS 服务器并验证（25 分钟）

```bash
# 用 Python 快速搭建 HTTPS 服务器
cat > https_server.py << 'PYEOF'
import http.server, ssl, json

class Handler(http.server.BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-Type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps({
            "message": "HTTPS server running",
            "path": self.path,
            "headers": dict(self.headers)
        }, indent=2).encode())

server = http.server.HTTPServer(('0.0.0.0', 8443), Handler)
ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
ctx.load_cert_chain('server.crt', 'server.key')
# 仅允许 TLS 1.2 和 1.3
ctx.minimum_version = ssl.TLSVersion.TLSv1_2
server.socket = ctx.wrap_socket(server.socket, server_side=True)
print("HTTPS server on https://0.0.0.0:8443")
server.serve_forever()
PYEOF

# 终端 1：启动服务器
python3 https_server.py

# 终端 2：测试连接
curl -k -v https://localhost:8443/

# 验证 TLS 版本和密码套件
echo | openssl s_client -connect localhost:8443 2>/dev/null | grep -E "Protocol|Cipher"

# 故意用不安全的 TLS 1.1 连接（应被拒绝）
curl --tlsv1.1 -k https://localhost:8443/ 2>&amp;1 | head -5
```

**检查标准**：
- [ ] 能从证书输出中识别 Issuer、Subject、SAN、有效期等字段
- [ ] 能独立生成 CA 根证书并签发服务器证书
- [ ] 能用 Python/Go 搭建自签名 HTTPS 服务器

### 步骤 1：DNS 解析全过程追踪（15 分钟）

```bash

# 2. 顶级域名服务器（.com）返回权威服务器地址

dig www.baidu.com AAAA       # IPv6 地址
dig www.baidu.com MX         # 邮件服务器
dig www.baidu.com NS         # 域名服务器
dig www.baidu.com TXT        # 文本记录（SPF/DKIM等）
dig www.baidu.com CNAME      # 别名记录
dig www.baidu.com ANY        # 所有记录

# 查看 TTL 值
dig www.baidu.com | grep -E "^[^;].*IN"
# 注意每条记录中的 TTL 字段（秒数）
```

## 练习五：WebSocket 实战（预计 60 分钟）

**目标**：理解 WebSocket 握手过程，能搭建 WebSocket 服务器并实现心跳检测、断线重连等工程实践。

### 步骤 1：观察 WebSocket 握手（15 分钟）

```bash

echo "Sec-WebSocket-Key: $WS_KEY"

# 发送 Upgrade 请求
curl -i -N \
  -H "Connection: Upgrade" \

  http://echo.websocket.org/ 2>&amp;1 | head -20

# 观察响应：
# - 101 Switching Protocols（状态码）
# - Upgrade: websocket（确认升级）
# - Sec-WebSocket-Accept（SHA-1 验证值）
```

### 步骤 2：搭建 WebSocket 服务器（25 分钟）

```bash
# 安装依赖
cd /tmp &amp;&amp; mkdir ws-practice &amp;&amp; cd ws-practice
npm init -y &amp;&amp; npm install ws

# 创建 WebSocket 服务器
cat > server.js << 'JSEOF'
const WebSocket = require('ws');
const http = require('http');

const server = http.createServer();
const wss = new WebSocket.Server({ server });

// 心跳检测配置
const HEARTBEAT_INTERVAL = 30000;  // 30秒发送一次 ping
const HEARTBEAT_TIMEOUT = 10000;   // 10秒内未收到 pong 则断开

wss.on('connection', (ws, req) => {
    console.log(`[连接] 新客户端接入, IP: ${req.socket.remoteAddress}`);
    ws.isAlive = true;
    ws.messageCount = 0;

    // 心跳响应
    ws.on('pong', () => { ws.isAlive = true; });

    // 接收消息
    ws.on('message', (data) => {
        ws.messageCount++;
        const msg = data.toString();
        console.log(`[消息 #${ws.messageCount}] ${msg}`);

        // Echo 回显
        ws.send(JSON.stringify({
            type: 'echo',
            original: msg,
            timestamp: Date.now(),
            messageCount: ws.messageCount
        }));
    });

    ws.on('close', (code, reason) => {
        console.log(`[断开] code=${code}, reason=${reason || 'none'}`);
    });

    ws.on('error', (err) => {
        console.error(`[错误] ${err.message}`);
    });
});

// 心跳定时器
const heartbeat = setInterval(() => {
    wss.clients.forEach((ws) => {
        if (!ws.isAlive) {
            console.log(`[心跳超时] 断开未响应客户端`);
            return ws.terminate();
        }
        ws.isAlive = false;
        ws.ping();
    });
}, HEARTBEAT_INTERVAL);

wss.on('close', () => clearInterval(heartbeat));

server.listen(8080, () => {
    console.log('WebSocket 服务器运行在 ws://localhost:8080');
    console.log(`心跳间隔: ${HEARTBEAT_INTERVAL}ms, 超时: ${HEARTBEAT_TIMEOUT}ms`);
});
JSEOF

# 终端 1：启动服务器
node server.js
```

### 步骤 3：客户端测试与断线重连（20 分钟）

```bash
# 创建测试客户端（带断线重连）
cat > client.js << 'JSEOF'
const WebSocket = require('ws');

const RECONNECT_DELAY = 3000;  // 重连间隔 3 秒
const MAX_RECONNECT = 5;       // 最大重连次数
let reconnectCount = 0;

function connect() {
    const ws = new WebSocket('ws://localhost:8080');

    ws.on('open', () => {
        console.log('[连接成功]');
        reconnectCount = 0;  // 重置重连计数

        // 发送测试消息
        let count = 0;
        const interval = setInterval(() => {
            if (ws.readyState === WebSocket.OPEN) {
                count++;
                ws.send(JSON.stringify({ id: count, data: `消息 ${count}` }));
            }
        }, 2000);

        ws.on('close', () => clearInterval(interval));
    });

    ws.on('message', (data) => {
        const msg = JSON.parse(data.toString());
        console.log(`[收到] 序列#${msg.messageCount}, 时间戳: ${msg.timestamp}`);
    });

    ws.on('close', (code, reason) => {
        console.log(`[断开] code=${code}, 5秒后尝试重连...`);
        if (reconnectCount < MAX_RECONNECT) {
            reconnectCount++;
            setTimeout(connect, RECONNECT_DELAY);
        } else {
            console.log('[放弃] 超过最大重连次数');
        }
    });

    ws.on('error', (err) => {
        console.error(`[错误] ${err.message}`);
    });
}

connect();
JSEOF

# 终端 2：启动客户端
node client.js

# 观察：
# 1. 消息收发的序列号递增
# 2. 服务端心跳 ping/pong 检测
# 3. 杀死服务端后客户端自动重连
```

**检查标准**：
- [ ] 能解释 WebSocket 握手中 Sec-WebSocket-Key/Accept 的验证原理
- [ ] 能搭建带心跳检测的 WebSocket 服务器
- [ ] 能实现客户端断线重连逻辑

```go
// 服务端流式 RPC
func (s *server) SayHelloStream(req *pb.HelloRequest, stream pb.Greeter_SayHelloStreamServer) error {
    log.Printf("SayHelloStream: name=%s, count=%d", req.Name, req.Count)
    count := int(req.Count)
    if count <= 0 { count = 5 }
    for i := 1; i <= count; i++ {
        if err := stream.Send(&amp;pb.HelloReply{
            Message:   fmt.Sprintf("Hello #%d, %s!", i, req.Name),
            Sequence:  int32(i),
            Timestamp: time.Now().Format(time.RFC3339),
        }); err != nil {
            return err
        }
        time.Sleep(500 * time.Millisecond)  // 模拟处理延迟
    }
    return nil
}

// 双向流式 RPC
func (s *server) Bidirectional(stream pb.Greeter_BidirectionalServer) error {
    for {
        msg, err := stream.Recv()
        if err == io.EOF { return nil }
        if err != nil { return err }
        log.Printf("Bidirectional recv: user=%s, content=%s", msg.User, msg.Content)

        if err := stream.Send(&amp;pb.ChatMessage{
            User:      "server",
            Content:   fmt.Sprintf("[%s] said: %s", msg.User, msg.Content),
            Timestamp: time.Now().UnixMilli(),
        }); err != nil {
            return err
        }
    }
}
```

```bash

cat > log_collector.py << 'PYEOF'
import json, ssl, http.server, datetime, os, threading
from kafka import KafkaProducer

# Kafka 生产者（发送日志到 Kafka）
producer = KafkaProducer(
    bootstrap_servers='localhost:9092',
    value_serializer=lambda v: json.dumps(v).encode('utf-8')
)

# 日志级别队列（模拟 WebSocket 推送目标）
log_queue = []

class LogHandler(http.server.BaseHTTPRequestHandler):
    def do_POST(self):
        length = int(self.headers.get('Content-Length', 0))
        body = self.rfile.read(length)
        try:
            log_entry = json.loads(body)
            log_entry['received_at'] = datetime.datetime.now().isoformat()
            log_entry['source'] = self.headers.get('X-Service-Name', 'unknown')

            # 发送到 Kafka
            producer.send('app-logs', value=log_entry)
            producer.flush()

            # 缓存到本地队列
            log_queue.append(log_entry)
            if len(log_queue) > 1000:
                log_queue.pop(0)

            self.send_response(200)
            self.send_header('Content-Type', 'application/json')
            self.end_headers()
            self.wfile.write(json.dumps({"status": "ok"}).encode())
        except Exception as e:
            self.send_response(400)
            self.end_headers()
            self.wfile.write(json.dumps({"error": str(e)}).encode())

    def do_GET(self):
        # 返回最近 50 条日志
        self.send_response(200)
        self.send_header('Content-Type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps(log_queue[-50:]).encode())

    def log_message(self, format, *args):
        pass  # 静默 HTTP 日志

# 使用 TLS
server = http.server.HTTPServer(('0.0.0.0', 9443), LogHandler)
ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
ctx.load_cert_chain('/tmp/nginx-cert/cert.pem', '/tmp/nginx-cert/key.pem')
ctx.minimum_version = ssl.TLSVersion.TLSv1_2
server.socket = ctx.wrap_socket(server.socket, server_side=True)
print("Log collector API on https://0.0.0.0:9443")
server.serve_forever()
PYEOF

pip install kafka-python 2>/dev/null
```

```bash
const WebSocket = require('ws');
const https = require('https');
const fs = require('fs');
const { Kafka } = require('kafkajs');

// Kafka 消费者
const kafka = new Kafka({ clientId: 'log-pusher', brokers: ['localhost:9092'] });
const consumer = kafka.consumer({ groupId: 'log-viewers' });
const logBuffer = [];

// WebSocket 服务器
const wss = new WebSocket.Server({
    port: 8081,
    perMessageDeflate: false  // 禁用压缩以降低延迟
});

wss.on('connection', (ws) => {
    console.log('[WS] 新客户端连接');
    // 发送历史日志
    ws.send(JSON.stringify({ type: 'history', logs: logBuffer.slice(-50) }));
});

async function startConsumer() {
    await consumer.connect();
    await consumer.subscribe({ topic: 'app-logs', fromBeginning: false });

    await consumer.run({
        eachMessage: async ({ message }) => {
            const log = JSON.parse(message.value.toString());
            logBuffer.push(log);
            if (logBuffer.length > 1000) logBuffer.shift();

            // 广播给所有 WebSocket 客户端
            wss.clients.forEach((client) => {
                if (client.readyState === WebSocket.OPEN) {
                    client.send(JSON.stringify({ type: 'log', data: log }));
                }
            });
        }
    });
}

startConsumer().catch(console.error);
console.log('WebSocket log pusher on ws://localhost:8081');
```
